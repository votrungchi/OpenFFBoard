291int32_t CANPort::addCanFilter(CAN_FilterTypeDef sFilterConfig){
292 takeSemaphore();
293 int32_t lowestId = sFilterConfig.FilterFIFOAssignment == CAN_RX_FIFO0 ? 0 : slaveFilterStart;
294 int32_t highestId = sFilterConfig.FilterFIFOAssignment == CAN_RX_FIFO0 ? slaveFilterStart : 29;
295 int32_t foundId = -1;
296 for(uint8_t id = lowestId; id < highestId ; id++ ){
297 for(CAN_FilterTypeDef filter : canFilters){
298 if(filter.FilterFIFOAssignment == sFilterConfig.FilterFIFOAssignment && id == filter.FilterBank){
299 break;
300 }
301 }
302 foundId = id;
303 break;
304 }
305 if(foundId < highestId){
306 if (HAL_CAN_ConfigFilter(this->hcan, &sFilterConfig) == HAL_OK){
307 canFilters.push_back(sFilterConfig);
308 }
309 }
310 giveSemaphore();
311 return foundId;
312}

318void CANPort::removeCanFilter(uint8_t filterId){
319 semaphore.Take();
320 for (uint8_t i = 0; i < canFilters.size(); i++){
321 if(canFilters[i].FilterBank == filterId){
322 canFilters[i].FilterActivation = false;
323 HAL_CAN_ConfigFilter(this->hcan, &canFilters[i]);
324 canFilters.erase(canFilters.begin()+i);
325 break;
326 }
327 }
328 semaphore.Give();
329}

332CommandStatus CANPort::command(const ParsedCommand& cmd,std::vector<CommandReply>& replies){
333
334 switch(static_cast<CanPort_commands>(cmd.cmdId)){
335
336 case CanPort_commands::speed:
337 if(cmd.type == CMDtype::get){
338 replies.push_back(CommandReply(this->speedPreset));
339 }else if(cmd.type == CMDtype::set){
340 setSpeedPreset(cmd.val);
341 }else if(cmd.type == CMDtype::info){
342 for(uint8_t i = 0; i<SpeedNames.size();i++){
343 replies.push_back(CommandReply(SpeedNames[i] + ":" + std::to_string(i)));
344 }
345 }
346 break;
347
348 case CanPort_commands::send:
349 {
350 if(cmd.type == CMDtype::setat){
351 if(!active){
352 start(); // If port is not used activate port at first use
353 }
354 CAN_tx_msg msg;
355 memcpy(msg.data,&cmd.val,8);
356 msg.header.StdId = cmd.adr;
357 sendMessage(msg);
358 }else{
359 return CommandStatus::NOT_FOUND;
360 }
361 break;
362 }
363 default:
364 return CommandStatus::NOT_FOUND;
365 }
366 return CommandStatus::OK;
367}
