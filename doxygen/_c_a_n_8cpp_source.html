275bool CANPort::sendMessage(CAN_TxHeaderTypeDef *pHeader, uint8_t aData[],uint32_t *pTxMailbox){
276 if(this->silent)
277 setSilentMode(false);
278 uint32_t mailbox;
279 if(pTxMailbox == nullptr){
280 pTxMailbox = &mailbox;
281 }
282
283 if(!HAL_CAN_GetTxMailboxesFreeLevel(hcan) && HAL_GetTick() - lastSentTime > sendTimeout){
284 // Mailbox full and nothing has been sent successfully for some time. Abort previous requests if timeout reached
285 HAL_CAN_AbortTxRequest(hcan, txMailboxes);
286 }
287//
288 if(!HAL_CAN_GetTxMailboxesFreeLevel(hcan)){ // Only wait if no mailbox is free
289 isWaitingFlag = true;
290 if(!takeSemaphore(sendTimeout)){
291 isWaitingFlag = false;
292 return false;
293 }
294
295 }
296
297 //this->isTakenFlag = true;
298 if (HAL_CAN_AddTxMessage(this->hcan, pHeader, aData, pTxMailbox) != HAL_OK)
299 {
300 /* Transmission request Error */
301 if(isWaitingFlag)
302 giveSemaphore();
303 return false;
304 }
305 txMailboxes |= *pTxMailbox;
306// if(HAL_CAN_GetTxMailboxesFreeLevel(hcan)) // Give back semaphore immediately if mailboxes are still free
307// giveSemaphore();
308 return true;
309}

343int32_t CANPort::addCanFilter(CAN_FilterTypeDef sFilterConfig){
344 configSem.Take();
345 int32_t lowestId = 0;// sFilterConfig.FilterFIFOAssignment == CAN_RX_FIFO0 ? 0 : slaveFilterStart;
346 int32_t highestId = slaveFilterStart;// sFilterConfig.FilterFIFOAssignment == CAN_RX_FIFO0 ? slaveFilterStart : 29;
347 int32_t foundId = -1;
348
349 for(uint8_t id = lowestId; id < highestId ; id++ ){
350 bool foundExisting = false;
351 for(CAN_FilterTypeDef filter : canFilters){
352 if(id == filter.FilterBank
353// && filter.FilterIdHigh == sFilterConfig.FilterIdHigh && filter.FilterIdLow == sFilterConfig.FilterIdLow &&filter.FilterFIFOAssignment == sFilterConfig.FilterFIFOAssignment &&
354// && filter.FilterMaskIdHigh == sFilterConfig.FilterMaskIdHigh && filter.FilterMaskIdLow == sFilterConfig.FilterMaskIdLow
355// && filter.FilterMode == sFilterConfig.FilterMode && filter.FilterScale == sFilterConfig.FilterScale
356 )
357 {
358 foundExisting = true;
359 break;
360 }
361 }
362 foundId = id;
363 if(!foundExisting){
364 break;
365 }
366
367 }
368 if(foundId < highestId){
369 if(sFilterConfig.FilterBank == 0)
370 sFilterConfig.FilterBank = foundId;
371 if (HAL_CAN_ConfigFilter(this->hcan, &sFilterConfig) == HAL_OK){
372 canFilters.push_back(sFilterConfig);
373 }
374 }
375 configSem.Give();
376 return foundId;

397CommandStatus CANPort::command(const ParsedCommand& cmd,std::vector<CommandReply>& replies){
398
399 switch(static_cast<CanPort_commands>(cmd.cmdId)){
400
401 case CanPort_commands::speed:
402 if(cmd.type == CMDtype::get){
403 replies.emplace_back(this->speedPreset);
404 }else if(cmd.type == CMDtype::set){
405 setSpeedPreset(cmd.val);
406 }else if(cmd.type == CMDtype::info){
407 for(uint8_t i = 0; i<SpeedNames.size();i++){
408 replies.emplace_back(SpeedNames[i] + ":" + std::to_string(i));
409 }
410 }
411 break;
412
413 case CanPort_commands::send:
414 {
415 if(cmd.type == CMDtype::setat){
416 if(!active){
417 start(); // If port is not used activate port at first use
418 }
419 CAN_tx_msg msg;
420 memcpy(msg.data,&cmd.val,8);
421 header.StdId = cmd.adr;
422 msg.header = header;
423 sendMessage(msg);
424 }else{
425 return CommandStatus::NOT_FOUND;
426 }
427 break;
428 }
429 case CanPort_commands::len:
430 handleGetSet(cmd, replies, header.DLC);
431 header.DLC = std::min<uint32_t>(header.DLC,8);
432 break;
433 default:
434 return CommandStatus::NOT_FOUND;
435 }
436 return CommandStatus::OK;
437}
