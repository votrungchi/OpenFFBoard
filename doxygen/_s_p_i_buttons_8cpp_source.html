92void SPI_Buttons::setConfig(ButtonSourceConfig config){
93 config.numButtons = std::min<uint8_t>(this->maxButtons, config.numButtons);
94 this->conf = config;
95
96 spiPort.freeCsPin(this->spiConfig.cs);
97 OutputPin* newPin = spiPort.getCsPin(config.cs_num-1); // TODO update internal pin number if requested pin is blocked
98 if(newPin != nullptr){
99 this->spiConfig.cs = *newPin;
100
101 }
102 spiPort.reserveCsPin(this->spiConfig.cs);
103 // Setup presets
104 if(conf.mode == SPI_BtnMode::TM){
105 this->spiConfig.cspol = true;
106 this->conf.cutRight = true;
107 this->spiConfig.peripheral.CLKPolarity = SPI_POLARITY_LOW;
108
109 }else if(conf.mode == SPI_BtnMode::PISOSR){
110 this->spiConfig.cspol = false;
111 this->conf.cutRight = false;
112 this->spiConfig.peripheral.CLKPhase = SPI_PHASE_1EDGE;
113 this->spiConfig.peripheral.CLKPolarity = SPI_POLARITY_LOW;
114 }
115 spiPort.takeSemaphore();
116 spiPort.configurePort(&this->spiConfig.peripheral);
117 spiPort.giveSemaphore();
118 if(config.numButtons == 64){ // Special case
119 mask = 0xffffffffffffffff;
120 }else{
121 mask = (uint64_t)pow<uint64_t>(2,config.numButtons)-(uint64_t)1; // Must be done completely in 64 bit!
122 }
123 offset = 8 - (config.numButtons % 8);
124
125 // Thrustmaster uses extra bits for IDs
126 if(config.mode == SPI_BtnMode::TM){
127 bytes = 1+((config.numButtons+2)/8);
128 }else{
129 bytes = 1+((config.numButtons-1)/8);
130 }
131
132 this->btnnum = config.numButtons;
133}

153void SPI_Buttons::process(uint64_t* buf){
154 if(offset){
155 if(this->conf.cutRight){
156 *buf = *buf >> offset;
157 }else{
158 *buf = *buf & this->mask;
159 }
160 }
161 if(conf.invert)
162 *buf = (~*buf);
163 *buf = *buf & mask;
164}

166__attribute__((optimize("-Ofast")))
167uint8_t SPI_Buttons::readButtons(uint64_t* buf){
168 memcpy(buf,this->spi_buf,std::min<uint8_t>(this->bytes,8));
169 process(buf); // give back last buffer
170
171 if(spiPort.isTaken() || !ready)
172 return this->btnnum; // Don't wait.
173
174 // CS pin and semaphore managed by spi port
175 spiPort.receive_DMA(spi_buf, bytes, this);
176
177 return this->btnnum;
178}

188CommandStatus SPI_Buttons::command(const ParsedCommand& cmd,std::vector<CommandReply>& replies){
189
190 switch(static_cast<SPIButtons_commands>(cmd.cmdId)){
191 case SPIButtons_commands::btnnum:
192 if(cmd.type == CMDtype::set){
193 ButtonSourceConfig* c = this->getConfig();
194 c->numButtons = cmd.val;
195 this->setConfig(*c);
196 }else if(cmd.type == CMDtype::get){
197 replies.push_back(CommandReply(this->getBtnNum()));
198 }else{
199 return CommandStatus::ERR;
200 }
201 break;
202 case SPIButtons_commands::btnpol:
203 if(cmd.type == CMDtype::set){
204 ButtonSourceConfig* c = this->getConfig();
205 c->invert = cmd.val != 0;
206 this->setConfig(*c);
207 }else if(cmd.type == CMDtype::get){
208 replies.push_back(CommandReply(this->getConfig()->invert ? 1 : 0));
209 }else{
210 return CommandStatus::ERR;
211 }
212 break;
213 case SPIButtons_commands::btncut:
214 if(cmd.type == CMDtype::set){
215 ButtonSourceConfig* c = this->getConfig();
216 c->cutRight = cmd.val != 0;
217 this->setConfig(*c);
218 }else if(cmd.type == CMDtype::get){
219 replies.push_back(CommandReply(this->getConfig()->cutRight ? 1 : 0));
220 }else{
221 return CommandStatus::ERR;
222 }
223 break;
224 case SPIButtons_commands::mode:
225 if(cmd.type == CMDtype::set){
226 setMode((SPI_BtnMode)cmd.val);
227 }else if(cmd.type == CMDtype::get){
228 replies.push_back(CommandReply((uint8_t)this->conf.mode));
229 }else if(cmd.type == CMDtype::info){
230 replies.push_back(CommandReply(printModes()));
231 }else{
232 return CommandStatus::ERR;
233 }
234 break;
235
236 case SPIButtons_commands::cs:
237 if (handleGetSet(cmd, replies, this->conf.cs_num) == CommandStatus::OK ) {
238 setConfig(this->conf);
239 }
240 break;
241
242 default:
243 return CommandStatus::NOT_FOUND;
244 }
245
246 return CommandStatus::OK;
247}

249static ButtonSourceConfig decodeIntToConf(uint16_t config_int, uint16_t config_int_2){
250 ButtonSourceConfig c;
251 c.numButtons = config_int & 0x3F;
252 c.invert = (config_int >> 6) & 0x1;
253 c.cutRight = (config_int >> 7) & 0x1;
254 c.mode = SPI_BtnMode(config_int >> 8);
255 c.cs_num = static_cast<uint8_t>(config_int_2 & 0xF); // Leaving space for other use.
256 return c;
257}
258static std::tuple<uint16_t, uint16_t> encodeConfToInt(ButtonSourceConfig* c){
259 uint16_t val = c->numButtons & 0x3F;
260 val |= c->invert << 6;
261 val |= c->cutRight << 7;
262 val |= (uint8_t)c->mode << 8;
263 return { val, c->cs_num & 0xF };
264}
