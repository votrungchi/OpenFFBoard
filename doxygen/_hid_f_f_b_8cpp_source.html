96void HidFFB::hidOut(uint8_t report_id, hid_report_type_t report_type, uint8_t const* buffer, uint16_t bufsize){
97 hidPeriodAvg.addValue((uint32_t)(HAL_GetTick() - lastOut)); // use uint16_t for timer overflow handling if micros timer is used
98 lastOut = HAL_GetTick();
99
100 // FFB Output Message
101 const uint8_t* report = buffer;
102 uint8_t event_idx = report_id - FFB_ID_OFFSET;
103
104 // -------- Out Reports --------
105 switch(event_idx)
106 {
107
108 case HID_ID_NEWEFREP: //add Effect Report. Feature
109 new_effect((FFB_CreateNewEffect_Feature_Data_t*)(report));
110 break;
111 case HID_ID_EFFREP: // Set Effect
112 {
113 FFB_SetEffect_t setEffectRepBuf;
114 memcpy(&setEffectRepBuf,report,std::min<uint16_t>(sizeof(FFB_SetEffect_t),bufsize)); // Copy report to buffer. only valid range if less axes are used
115 set_effect(&setEffectRepBuf);
116 break;
117 }
118 case HID_ID_CTRLREP: // Control report. 1=Enable Actuators, 2=Disable Actuators, 4=Stop All Effects, 8=Reset, 16=Pause, 32=Continue
119 ffb_control(report[1]);
120 //sendStatusReport(0);
121 break;
122 case HID_ID_GAINREP: // Set global gain
123 set_gain(report[1]);
124 break;
125 case HID_ID_ENVREP: // Envelope
126 set_envelope((FFB_SetEnvelope_Data_t *)report);
127 break;
128 case HID_ID_CONDREP: // Spring, Damper, Friction, Inertia
129 set_condition((FFB_SetCondition_Data_t*)report);
130 break;
131 case HID_ID_PRIDREP: // Periodic
132 set_periodic((FFB_SetPeriodic_Data_t*)report);
133 break;
134 case HID_ID_CONSTREP: // Constant
135 set_constant_effect((FFB_SetConstantForce_Data_t*)report);
136 break;
137 case HID_ID_RAMPREP: // Ramp
138 set_ramp((FFB_SetRamp_Data_t *)report);
139 break;
140 case HID_ID_CSTMREP: // Custom. pretty much never used
141 //printf("Customrep");
142 break;
143 case HID_ID_SMPLREP: // Download sample
144 //printf("Sampledl");
145 break;
146 case HID_ID_EFOPREP: //Effect operation
147 {
148 set_effect_operation((FFB_EffOp_Data_t*)report);
149 break;
150 }
151 case HID_ID_BLKFRREP: // Free a block
152 {
153 free_effect(report[1]-1);
154 break;
155 }
156
157 default:
158 {
159 break;
160 }
161 }

184uint16_t HidFFB::hidGet(uint8_t report_id, hid_report_type_t report_type,uint8_t* buffer, uint16_t reqlen){
185 // Feature gets go here
186
187 uint8_t id = report_id - FFB_ID_OFFSET;
188
189 switch(id){
190 case HID_ID_BLKLDREP:
191 //printf("Get Block Report\n");
192 // Notice: first byte ID is not present in the reply buffer because it is handled by tinyusb internally!
193 memcpy(buffer,&this->blockLoad_report,sizeof(FFB_BlockLoad_Feature_Data_t));
194 return sizeof(FFB_BlockLoad_Feature_Data_t);
195 break;
196 case HID_ID_POOLREP:
197 //printf("Get Pool Report\n");
198 memcpy(buffer,&this->pool_report,sizeof(FFB_PIDPool_Feature_Data_t));
199 return sizeof(FFB_PIDPool_Feature_Data_t);
200 break;
201 default:
202 break;
203 }
204 return 0;
205}

274void HidFFB::new_effect(FFB_CreateNewEffect_Feature_Data_t* effect){
275 // Allocates a new effect
276
277 uint8_t index = find_free_effect(effect->effectType); // next effect
278 if(index == 0){
279 blockLoad_report.loadStatus = 2;
280#ifdef DEBUGLOG
281 CommandHandler::logSerialDebug("Can't allocate a new effect");
282#endif
283 return;
284 }
285 FFB_Effect new_effect;
286 new_effect.type = effect->effectType;
287
288 this->effects_calc->logEffectType(effect->effectType,false);
289#ifdef DEBUGLOG
290 CommandHandler::logSerialDebug("New effect type:" + std::to_string(effect->effectType) + " idx: " + std::to_string(index-1));
291#endif
292
293 set_filters(&new_effect);
294
295 effects[index-1] = std::move(new_effect);
296 // Set block load report
297 //reportFFBStatus.effectBlockIndex = index;
298 blockLoad_report.effectBlockIndex = index;
299 used_effects++;
300 blockLoad_report.ramPoolAvailable = (effects.size()-used_effects)*sizeof(FFB_Effect);
301 blockLoad_report.loadStatus = 1;
302 sendStatusReport(index);
303

306void HidFFB::set_effect(FFB_SetEffect_t* effect){
307 uint8_t index = effect->effectBlockIndex;
308 if(index > effects.size() || index == 0)

310

321

323 effect_p->directionX = effect->directionX;
324 effect_p->directionY = effect->directionY;
325#if MAX_AXIS == 3
326 effect_p->directionZ = effect->directionZ;
327#endif
328 if(effect->duration == 0){ // Fix for games assuming 0 is infinite
329 effect_p->duration = FFB_EFFECT_DURATION_INFINITE;
330 }else{
331 effect_p->duration = effect->duration;

355 effect->conditions[axis].deadBand = cond->deadBand;
356 //effect->conditionsCount++;

358 effect->conditions[axis].positiveSaturation = 0x7FFF;

365void HidFFB::set_effect_operation(FFB_EffOp_Data_t* report){
366 if(report->effectBlockIndex == 0 || report->effectBlockIndex > effects.size()){
367 return; // Invalid ID
368 }
369 // Start or stop effect
370 uint8_t id = report->effectBlockIndex-1;
371 if(report->state == 3){
372 effects[id].state = 0; //Stop
373#ifdef DEBUGLOG
374 CommandHandler::logSerialDebug("Stop effect: " + std::to_string(id));
375#endif
376
377 }else{
378
379 // 1 = start, 2 = start solo
380 if(report->state == 2){
381#ifdef DEBUGLOG
382 CommandHandler::logSerialDebug("Start solo: " + std::to_string(id));
383#endif
384 for(FFB_Effect& effect : effects){
385 effect.state = 0; // Stop all other effects
386 }
387 }
388 if(effects[id].state != 1){
389 set_filters(&effects[id]);
390 }
391#ifdef DEBUGLOG
392 CommandHandler::logSerialDebug("Start effect: " + std::to_string(id));
393#endif
394 effects[id].startTime = HAL_GetTick() + effects[id].startDelay; // + effects[id].startDelay;
395 effects[id].state = 1; //Start
396
397
398 }
399 //sendStatusReport(report[1]);
400 this->effects_calc->logEffectState(effects[id].type,effects[id].state);
401}
