86void HidFFB::hidOut(uint8_t report_id, hid_report_type_t report_type, uint8_t const* buffer, uint16_t bufsize){
87 hid_out_period = (micros() - lastOut); // For measuring update rate
88 lastOut = micros();
89 // FFB Output Message
90 const uint8_t* report = buffer;
91 uint8_t event_idx = report_id - FFB_ID_OFFSET;
92
93
94 // -------- Out Reports --------
95 switch(event_idx)
96 {
97
98 case HID_ID_NEWEFREP: //add Effect Report. Feature
99 new_effect((FFB_CreateNewEffect_Feature_Data_t*)(report));
100 break;
101 case HID_ID_EFFREP: // Set Effect
102 {
103 FFB_SetEffect_t setEffectRepBuf;
104 memcpy(&setEffectRepBuf,report,std::min<uint16_t>(sizeof(FFB_SetEffect_t),bufsize)); // Copy report to buffer. only valid range if less axes are used
105 set_effect(&setEffectRepBuf);
106 break;
107 }
108 case HID_ID_CTRLREP: // Control report. 1=Enable Actuators, 2=Disable Actuators, 4=Stop All Effects, 8=Reset, 16=Pause, 32=Continue
109 ffb_control(report[1]);
110 //sendStatusReport(0);
111 break;
112 case HID_ID_GAINREP: // Set global gain
113 set_gain(report[1]);
114 break;
115 case HID_ID_ENVREP: // Envelope
116 set_envelope((FFB_SetEnvelope_Data_t *)report);
117 break;
118 case HID_ID_CONDREP: // Spring, Damper, Friction, Inertia
119 set_condition((FFB_SetCondition_Data_t*)report);
120 break;
121 case HID_ID_PRIDREP: // Periodic
122 set_periodic((FFB_SetPeriodic_Data_t*)report);
123 break;
124 case HID_ID_CONSTREP: // Constant
125 set_constant_effect((FFB_SetConstantForce_Data_t*)report);
126 break;
127 case HID_ID_RAMPREP: // Ramp
128 set_ramp((FFB_SetRamp_Data_t *)report);
129 break;
130 case HID_ID_CSTMREP: // Custom. pretty much never used
131 //printf("Customrep");
132 break;
133 case HID_ID_SMPLREP: // Download sample
134 //printf("Sampledl");
135 break;
136 case HID_ID_EFOPREP: //Effect operation
137 {
138 set_effect_operation((FFB_EffOp_Data_t*)report);
139 break;
140 }
141 case HID_ID_BLKFRREP: // Free a block
142 {
143 free_effect(report[1]-1);
144 break;
145 }
146
147 default:
148 {
149 break;
150 }
151 }

173uint16_t HidFFB::hidGet(uint8_t report_id, hid_report_type_t report_type,uint8_t* buffer, uint16_t reqlen){
174 // Feature gets go here
175
176 uint8_t id = report_id - FFB_ID_OFFSET;
177
178 switch(id){
179 case HID_ID_BLKLDREP:
180 //printf("Get Block Report\n");
181 // Notice: first byte ID is not present in the reply buffer because it is handled by tinyusb internally!
182 memcpy(buffer,&this->blockLoad_report,sizeof(FFB_BlockLoad_Feature_Data_t));
183 return sizeof(FFB_BlockLoad_Feature_Data_t);
184 break;
185 case HID_ID_POOLREP:
186 //printf("Get Pool Report\n");
187 memcpy(buffer,&this->pool_report,sizeof(FFB_PIDPool_Feature_Data_t));
188 return sizeof(FFB_PIDPool_Feature_Data_t);
189 break;
190 default:
191 break;
192 }
193 return 0;
194}

236 stop_FFB();

255void HidFFB::new_effect(FFB_CreateNewEffect_Feature_Data_t* effect){
256 // Allocates a new effect
257
258 uint8_t index = find_free_effect(effect->effectType); // next effect
259 if(index == 0){
260 blockLoad_report.loadStatus = 2;
261 //CommandHandler::logSerial("Can't allocate a new effect");
262 return;
263 }
264 FFB_Effect new_effect;
265 new_effect.type = effect->effectType;
266 this->effects_calc->logEffectType(effect->effectType);
267
268 set_filters(&new_effect);
269
270 effects[index-1] = std::move(new_effect);
271 // Set block load report
272 //reportFFBStatus.effectBlockIndex = index;
273 blockLoad_report.effectBlockIndex = index;
274 used_effects++;
275 blockLoad_report.ramPoolAvailable = (MAX_EFFECTS-used_effects)*sizeof(FFB_Effect);
276 blockLoad_report.loadStatus = 1;
277 sendStatusReport(index);
278

281void HidFFB::set_effect(FFB_SetEffect_t* effect){
282 uint8_t index = effect->effectBlockIndex;
283 if(index > MAX_EFFECTS || index == 0)
284 return;
285
286 FFB_Effect* effect_p = &effects[index-1];
287
288 if (effect_p->type != effect->effectType){
289 effect_p->startTime = 0;
290 set_filters(effect_p);
291 }
292
293 effect_p->gain = effect->gain;
294 effect_p->type = effect->effectType;
295 effect_p->samplePeriod = effect->samplePeriod;
296
297 effect_p->enableAxis = effect->enableAxis;
298 effect_p->directionX = effect->directionX;
299 effect_p->directionY = effect->directionY;
300#if MAX_AXIS == 3
301 effect_p->directionZ = effect->directionZ;
302#endif
303 if(effect->duration == 0){ // Fix for games assuming 0 is infinite
304 effect_p->duration = FFB_EFFECT_DURATION_INFINITE;
305 }else{
306 effect_p->duration = effect->duration;
307 }
308 effect_p->startDelay = effect->startDelay;
309 if(!ffb_active)
310 start_FFB();
311 sendStatusReport(effect->effectBlockIndex); // TODO required?
312 //CommandHandler::logSerial("Setting Effect: " + std::to_string(effect->effectType) + " at " + std::to_string(index) + "\n");

339void HidFFB::set_effect_operation(FFB_EffOp_Data_t* report){
340 if(report->effectBlockIndex == 0 || report->effectBlockIndex > MAX_EFFECTS){
341 return; // Invalid ID
342 }
343 // Start or stop effect
344 uint8_t id = report->effectBlockIndex-1;
345 if(report->state == 3){
346 effects[id].state = 0; //Stop
347 //CommandHandler::logSerial("Stop" + std::to_string(id));
348 }else{
349
350 // 1 = start, 2 = start solo
351 if(report->state == 2){
352 //CommandHandler::logSerial("Start solo" + std::to_string(id));
353 for(FFB_Effect& effect : effects){
354 effect.state = 0; // Stop all other effects
355 }
356 }
357 if(effects[id].state != 1){
358 set_filters(&effects[id]);
359 }
360
361 //CommandHandler::logSerial("Start" + std::to_string(id));
362 effects[id].startTime = HAL_GetTick() + effects[id].startDelay; // + effects[id].startDelay;
363 effects[id].state = 1; //Start
364
365
366 }
367 //sendStatusReport(report[1]);
