58void MotorPWM::turn(int16_t power){
59 if(!active)
60 return;
61
62 /*
63 * Generates a classic RC 20ms 1000-2000µs signal
64 * Centered at 1500µs for bidirectional RC ESCs and similiar stuff
65 */
66 if(mode == ModePWM_DRV::RC_PWM){
67 int32_t pval = power;
68
69 float val = ((pval * 1000)/0x7fff)*tFreq;
70 val = clip((1500*tFreq)-val,1000*tFreq, 2000*tFreq);
71 setPWM(val,timerConfig.ccr_1);
72
73 /*
74 * Generates a 0-100% PWM signal
75 * and outputs complementary direction signals.
76 * Can be used with cheap halfbridge modules and DC motors
77 */
78 }else if(mode == ModePWM_DRV::PWM_DIR){
79 if(power < 0){
80 setPWM(0,timerConfig.ccr_3);
81 setPWM(0xffff,timerConfig.ccr_4);
82 }else{
83 setPWM(0,timerConfig.ccr_4);
84 setPWM(0xffff,timerConfig.ccr_3);
85 }
86 int32_t val = (uint32_t)((abs(power) * period)/0x7fff);
87 setPWM(val,timerConfig.ccr_1);
88
89 }else if(mode == ModePWM_DRV::CENTERED_PWM){
90 int32_t pval = 0x7fff+power;
91 int32_t val = (pval * period)/0xffff;
92 setPWM(val,timerConfig.ccr_1);
93
94 }else if(mode == ModePWM_DRV::PWM_DUAL){
95 int32_t val = (uint32_t)((abs(power) * period)/0x7fff);
96 if(power < 0){
97 setPWM(0,timerConfig.ccr_1);
98 setPWM(val,timerConfig.ccr_2);
99 }else{
100 setPWM(0,timerConfig.ccr_2);
101 setPWM(val,timerConfig.ccr_1);
102 }
103 }

109void MotorPWM::setPwmSpeed(SpeedPWM_DRV spd){
110 bool ok = true;
111 switch(spd){
112
113 case SpeedPWM_DRV::LOW:
114 if(mode == ModePWM_DRV::RC_PWM){
115 period = 40000; //20ms (40000/Sysclock)
116 prescaler = timerConfig.timerFreq/2000000;
117 }else{
118 period = timerConfig.timerFreq/3000; // Check if timer can count high enough for very high clock speeds!
119 prescaler = 0;
120 }
121
122 break;
123 case SpeedPWM_DRV::MID:
124 if(mode == ModePWM_DRV::RC_PWM){
125 period = 30000;//15ms(30000/47)
126 prescaler = timerConfig.timerFreq/2000000;
127 }else{
128 period = timerConfig.timerFreq/9000;
129 prescaler = 0;
130 }
131
132 break;
133 case SpeedPWM_DRV::HIGH:
134 if(mode == ModePWM_DRV::RC_PWM){
135 period = 20000; //10ms (20000/47)
136 prescaler = timerConfig.timerFreq/2000000;
137 }else{
138 period = timerConfig.timerFreq/17000;
139 prescaler = 0;
140 }
141 break;
142 case SpeedPWM_DRV::VERYHIGH:
143 if(mode == ModePWM_DRV::RC_PWM){
144 period = 10000; //5ms (20000/23)
145 prescaler = timerConfig.timerFreq/2000000;
146 }else{
147 period = timerConfig.timerFreq/24000;
148 prescaler = 0;
149 }
150 break;
151 default:
152 ok = false;
153 }
154
155 if(ok){
156 this->pwmspeed = spd;
157 tFreq = (float)(timerConfig.timerFreq/1000000)/(float)(prescaler+1);
158
159 pwmInitTimer(timerConfig.timer, timerConfig.channel_1,period,prescaler);
160 pwmInitTimer(timerConfig.timer, timerConfig.channel_2,period,prescaler);
161 pwmInitTimer(timerConfig.timer, timerConfig.channel_3,period,prescaler);
162 pwmInitTimer(timerConfig.timer, timerConfig.channel_4,period,prescaler);
163 HAL_TIM_MspPostInit(timerConfig.timer);
164// setPWM_HAL(0, timer, channel_1, period);
165// pwmInitTimer(timer, channel_2,period,prescaler);
166// setPWM_HAL(0, timer, channel_2, period);
167 turn(0);
168 }
169}

194void MotorPWM::saveFlash(){
195 // 0-3: mode
196 // 4-6: speed
197 uint16_t var = (uint8_t)this->mode & 0xf;
198 var |= ((uint8_t)this->pwmspeed & 0x7) << 4;
199 Flash_Write(ADR_PWM_MODE, var);
200}
201void MotorPWM::restoreFlash(){
202 uint16_t var = 0;
203 if(Flash_Read(ADR_PWM_MODE, &var)){
204 uint8_t m = var & 0xf;
205 this->setMode(ModePWM_DRV(m));
206
207 uint8_t s = (var >> 4) & 0x7;
208 this->setPwmSpeed(SpeedPWM_DRV(s));
209 }
210}

235CommandStatus MotorPWM::command(const ParsedCommand& cmd,std::vector<CommandReply>& replies){
236
237 switch(static_cast<MotorPWM_commands>(cmd.cmdId)){
238
239 case MotorPWM_commands::freq:
240 {
241 if(cmd.type == CMDtype::set){
242 this->setPwmSpeed((SpeedPWM_DRV)cmd.val);
243 }else if(cmd.type == CMDtype::get){
244 replies.emplace_back((uint8_t)this->getPwmSpeed());
245 }else if(cmd.type == CMDtype::info){
246 std::vector<std::string> names = PWM_SpeedNames;
247 if(this->mode == ModePWM_DRV::RC_PWM){
248 names = RC_SpeedNames;
249 }else{
250 names = PWM_SpeedNames;
251 }
252 for(uint8_t i = 0; i<names.size();i++){
253 replies.emplace_back(names[i] + ":" + std::to_string(i)+"\n");
254 }
255 }
256 break;
257 }
258 case MotorPWM_commands::mode:
259 {
260 if(cmd.type == CMDtype::set){
261 this->setMode((ModePWM_DRV)cmd.val);
262 }else if(cmd.type == CMDtype::get){
263 replies.emplace_back((uint8_t)this->getMode());
264 }else if(cmd.type == CMDtype::info){
265 for(uint8_t i = 0; i<PwmModeNames.size();i++){
266 replies.emplace_back(PwmModeNames[i] + ":" + std::to_string(i)+"\n");
267 }
268 }
269 break;
270 }
271 default:
272 return CommandStatus::NOT_FOUND;
273 }
274
275 return CommandStatus::OK;
276
277}

293 pwmTimMutex.Lock();

295 HAL_TIM_PWM_ConfigChannel(timer, &sConfigOC, channel);
