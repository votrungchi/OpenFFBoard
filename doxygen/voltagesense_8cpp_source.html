50void brakeCheck(){
51 if(maxVoltage == 0 || brake_failure){
52 return;
53 }
54 uint16_t vint = getIntV();
55 uint16_t vext = getExtV();
56 bool lastBrakingFlag = braking_flag;
57 braking_flag = (vint > vext + voltageDiffActivate) || vint > maxVoltage || (braking_flag && (vint > vext + voltageDiffDeactivate));
58 //(ADC_BUF[ADC_CHAN_VINT] > ADC_BUF[ADC_CHAN_VEXT]+400 || (ADC_BUF[ADC_CHAN_VINT] > 3000));
59 if(braking_flag != lastBrakingFlag){
60 if(braking_flag){
61 brakeActiveTime = HAL_GetTick();
62 }else{
63 brakeActiveTime = 0;
64 }
65 }
66 if(brakeActiveTime && HAL_GetTick() - brakeActiveTime > 5000){
67 // Brake resistor active over 5s. Shut down.
68 brake_failure = true;
69 HAL_GPIO_WritePin(DRV_BRAKE_GPIO_Port,DRV_BRAKE_Pin, GPIO_PIN_RESET);
70 ErrorHandler::addError(Error(ErrorCode::brakeResistorFailure, ErrorType::critical, "Brake resistor stuck on"));
71 }else{
72 HAL_GPIO_WritePin(DRV_BRAKE_GPIO_Port,DRV_BRAKE_Pin, braking_flag ? GPIO_PIN_SET:GPIO_PIN_RESET);
73 }
