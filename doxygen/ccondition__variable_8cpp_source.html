77void ConditionVariable::Signal()
78{
79 //
80 // Lock the internal condition variable state
81 //
82 Lock.Lock();
83
84 if ( !WaitList.empty() ) {
85
86 Thread *thr = WaitList.front();
87 WaitList.pop_front();

91 //
92 // Drop the internal condition variable lock.
