27/*
28 * This library is not fully reentrant, though it is reentrant from the view
29 * of either the application layer or the USB stack. Due to its locking,
30 * it is not safe to call its functions from interrupts.
31 *
32 * The one exception is that its functions may not be called from the application
33 * until the USB stack is initialized. This should not be a problem since the
34 * device shouldn't be sending messages until it receives a request from the
35 * host.
36 */
37
38
39/*
40 * In the case of single-CPU "no OS", this task is never preempted other than by
41 * interrupts, and the USBTMC code isn't called by interrupts, so all is OK. For "no OS",
42 * the mutex structure's main effect is to disable the USB interrupts.
43 * With an OS, this class driver uses the OSAL to perform locking. The code uses a single lock
44 * and does not call outside of this class with a lock held, so deadlocks won't happen.
45 */
46
47//Limitations:
48// "vendor-specific" commands are not handled.
49// Dealing with "termchar" must be handled by the application layer,
50// though additional error checking is does in this module.
51// talkOnly and listenOnly are NOT supported. They're not permitted
52// in USB488, anyway.
53
54/* Supported:
55 *
56 * Notification pulse
57 * Trigger
58 * Read status byte (both by interrupt endpoint and control message)
59 *
60 */
61
62
63// TODO:
64// USBTMC 3.2.2 error conditions not strictly followed
65// No local lock-out, REN, or GTL.
66// Clear message available status byte at the correct time? (488 4.3.1.3)
67

83/*
84 * The state machine does not allow simultaneous reading and writing. This is
85 * consistent with USBTMC.
86 */
87
88typedef enum
89{
90 STATE_CLOSED, // Endpoints have not yet been opened since USB reset
91 STATE_NAK, // Bulk-out endpoint is in NAK state.
92 STATE_IDLE, // Bulk-out endpoint is waiting for CMD.
93 STATE_RCV, // Bulk-out is receiving DEV_DEP message
94 STATE_TX_REQUESTED,
95 STATE_TX_INITIATED,
96 STATE_TX_SHORTED,
97 STATE_CLEARING,
98 STATE_ABORTING_BULK_IN,
99 STATE_ABORTING_BULK_IN_SHORTED, // aborting, and short packet has been queued for transmission
100 STATE_ABORTING_BULK_IN_ABORTED, // aborting, and short packet has been transmitted
101 STATE_ABORTING_BULK_OUT,
102 STATE_NUM_STATES
103} usbtmcd_state_enum;

112typedef struct
113{
114 volatile usbtmcd_state_enum state;
115
116 uint8_t itf_id;
117 uint8_t rhport;
118 uint8_t ep_bulk_in;
119 uint8_t ep_bulk_out;
120 uint8_t ep_int_in;
121 // IN buffer is only used for first packet, not the remainder
122 // in order to deal with prepending header
123 CFG_TUSB_MEM_ALIGN uint8_t ep_bulk_in_buf[USBTMCD_MAX_PACKET_SIZE];
124 // OUT buffer receives one packet at a time
125 CFG_TUSB_MEM_ALIGN uint8_t ep_bulk_out_buf[USBTMCD_MAX_PACKET_SIZE];
126 uint32_t transfer_size_remaining; // also used for requested length for bulk IN.
127 uint32_t transfer_size_sent; // To keep track of data bytes that have been queued in FIFO (not header bytes)
128
129 uint8_t lastBulkOutTag; // used for aborts (mostly)
130 uint8_t lastBulkInTag; // used for aborts (mostly)
131
132 uint8_t const * devInBuffer; // pointer to application-layer used for transmissions
133
134 usbtmc_capabilities_specific_t const * capabilities;
135} usbtmc_interface_state_t;

158// Our own private lock, mostly for the state variable.
159#define criticalEnter() do {osal_mutex_lock(usbtmcLock,OSAL_TIMEOUT_WAIT_FOREVER); } while (0)
160#define criticalLeave() do {osal_mutex_unlock(usbtmcLock); } while (0)

162bool atomicChangeState(usbtmcd_state_enum expectedState, usbtmcd_state_enum newState)
163{
164 bool ret = true;
165 criticalEnter();
166 usbtmcd_state_enum oldState = usbtmc_state.state;
167 if (oldState == expectedState)
168 {
169 usbtmc_state.state = newState;
170 }
171 else
172 {
173 ret = false;
174 }
175 criticalLeave();
176 return ret;
177}

179// called from app
180// We keep a reference to the buffer, so it MUST not change until the app is
181// notified that the transfer is complete.
182// length of data is specified in the hdr.
183
184// We can't just send the whole thing at once because we need to concatanate the
185// header with the data.
186bool tud_usbtmc_transmit_dev_msg_data(
187 const void * data, size_t len,
188 bool endOfMessage,
189 bool usingTermChar)
190{
191 const unsigned int txBufLen = sizeof(usbtmc_state.ep_bulk_in_buf);
192
193#ifndef NDEBUG
194 TU_ASSERT(len > 0u);
195 TU_ASSERT(len <= usbtmc_state.transfer_size_remaining);
196 TU_ASSERT(usbtmc_state.transfer_size_sent == 0u);
197 if(usingTermChar)
198 {
199 TU_ASSERT(usbtmc_state.capabilities->bmDevCapabilities.canEndBulkInOnTermChar);
200 TU_ASSERT(termCharRequested);
201 TU_ASSERT(((uint8_t const*)data)[len-1u] == termChar);
202 }
203#endif
204
205 TU_VERIFY(usbtmc_state.state == STATE_TX_REQUESTED);
206 usbtmc_msg_dev_dep_msg_in_header_t *hdr = (usbtmc_msg_dev_dep_msg_in_header_t*)usbtmc_state.ep_bulk_in_buf;
207 tu_varclr(hdr);
208 hdr->header.MsgID = USBTMC_MSGID_DEV_DEP_MSG_IN;
209 hdr->header.bTag = usbtmc_state.lastBulkInTag;
210 hdr->header.bTagInverse = (uint8_t)~(usbtmc_state.lastBulkInTag);
211 hdr->TransferSize = len;
212 hdr->bmTransferAttributes.EOM = endOfMessage;
213 hdr->bmTransferAttributes.UsingTermChar = usingTermChar;
214
215 // Copy in the header
216 const size_t headerLen = sizeof(*hdr);
217 const size_t dataLen = ((headerLen + hdr->TransferSize) <= txBufLen) ?
218 len : (txBufLen - headerLen);
219 const size_t packetLen = headerLen + dataLen;
220
221 memcpy((uint8_t*)(usbtmc_state.ep_bulk_in_buf) + headerLen, data, dataLen);
222 usbtmc_state.transfer_size_remaining = len - dataLen;
223 usbtmc_state.transfer_size_sent = dataLen;
224 usbtmc_state.devInBuffer = (uint8_t const*) data + (dataLen);
225
226 bool stateChanged =
227 atomicChangeState(STATE_TX_REQUESTED, (packetLen >= txBufLen) ? STATE_TX_INITIATED : STATE_TX_SHORTED);
228 TU_VERIFY(stateChanged);
229 TU_VERIFY(usbd_edpt_xfer(usbtmc_state.rhport, usbtmc_state.ep_bulk_in, usbtmc_state.ep_bulk_in_buf, (uint16_t)packetLen));
230 return true;
231}

233void usbtmcd_init_cb(void)
234{
235 usbtmc_state.capabilities = tud_usbtmc_get_capabilities_cb();
236#ifndef NDEBUG
237# if CFG_TUD_USBTMC_ENABLE_488
238 if (usbtmc_state.capabilities->bmIntfcCapabilities488.supportsTrigger) {
239 TU_ASSERT(&tud_usbtmc_msg_trigger_cb != NULL,);
240 }
241 // Per USB488 spec: table 8
242 TU_ASSERT(!usbtmc_state.capabilities->bmIntfcCapabilities.listenOnly,);
243 TU_ASSERT(!usbtmc_state.capabilities->bmIntfcCapabilities.talkOnly,);
244# endif
245 if (usbtmc_state.capabilities->bmIntfcCapabilities.supportsIndicatorPulse) {
246 TU_ASSERT(&tud_usbtmc_indicator_pulse_cb != NULL,);
247 }
248#endif
249
250 usbtmcLock = osal_mutex_create(&usbtmcLockBuffer);
251}

253uint16_t usbtmcd_open_cb(uint8_t rhport, tusb_desc_interface_t const * itf_desc, uint16_t max_len)
254{
255 (void)rhport;
256
257 uint16_t drv_len;
258 uint8_t const * p_desc;
259 uint8_t found_endpoints = 0;
260
261 TU_VERIFY(itf_desc->bInterfaceClass == TUD_USBTMC_APP_CLASS , 0);
262 TU_VERIFY(itf_desc->bInterfaceSubClass == TUD_USBTMC_APP_SUBCLASS, 0);
263
264#ifndef NDEBUG
265 // Only 2 or 3 endpoints are allowed for USBTMC.
266 TU_ASSERT((itf_desc->bNumEndpoints == 2) || (itf_desc->bNumEndpoints ==3), 0);
267#endif
268
269 TU_ASSERT(usbtmc_state.state == STATE_CLOSED, 0);
270
271 // Interface
272 drv_len = 0u;
273 p_desc = (uint8_t const *) itf_desc;
274
275 usbtmc_state.itf_id = itf_desc->bInterfaceNumber;
276 usbtmc_state.rhport = rhport;
277
278 while (found_endpoints < itf_desc->bNumEndpoints && drv_len <= max_len)
279 {
280 if ( TUSB_DESC_ENDPOINT == p_desc[DESC_OFFSET_TYPE])
281 {
282 tusb_desc_endpoint_t const *ep_desc = (tusb_desc_endpoint_t const *)p_desc;
283 switch(ep_desc->bmAttributes.xfer) {
284 case TUSB_XFER_BULK:
285 TU_ASSERT(tu_edpt_packet_size(ep_desc) == USBTMCD_MAX_PACKET_SIZE, 0);
286 if (tu_edpt_dir(ep_desc->bEndpointAddress) == TUSB_DIR_IN)
287 {
288 usbtmc_state.ep_bulk_in = ep_desc->bEndpointAddress;
289 } else {
290 usbtmc_state.ep_bulk_out = ep_desc->bEndpointAddress;
291 }
292
293 break;
294 case TUSB_XFER_INTERRUPT:
295#ifndef NDEBUG
296 TU_ASSERT(tu_edpt_dir(ep_desc->bEndpointAddress) == TUSB_DIR_IN, 0);
297 TU_ASSERT(usbtmc_state.ep_int_in == 0, 0);
298#endif
299 usbtmc_state.ep_int_in = ep_desc->bEndpointAddress;
300 break;
301 default:
302 TU_ASSERT(false, 0);
303 }
304 TU_ASSERT( usbd_edpt_open(rhport, ep_desc), 0);
305 found_endpoints++;
306 }
307
308 drv_len += tu_desc_len(p_desc);
309 p_desc = tu_desc_next(p_desc);
310 }
311
312 // bulk endpoints are required, but interrupt IN is optional
313#ifndef NDEBUG
314 TU_ASSERT(usbtmc_state.ep_bulk_in != 0, 0);
315 TU_ASSERT(usbtmc_state.ep_bulk_out != 0, 0);
316 if (itf_desc->bNumEndpoints == 2)
317 {
318 TU_ASSERT(usbtmc_state.ep_int_in == 0, 0);
319 }
320 else if (itf_desc->bNumEndpoints == 3)
321 {
322 TU_ASSERT(usbtmc_state.ep_int_in != 0, 0);
323 }
324#if (CFG_TUD_USBTMC_ENABLE_488)
325 if(usbtmc_state.capabilities->bmIntfcCapabilities488.is488_2 ||
326 usbtmc_state.capabilities->bmDevCapabilities488.SR1)
327 {
328 TU_ASSERT(usbtmc_state.ep_int_in != 0, 0);
329 }
330#endif
331#endif
332 atomicChangeState(STATE_CLOSED, STATE_NAK);
333 tud_usbtmc_open_cb(itf_desc->iInterface);
334
335 return drv_len;
336}
337// Tell USBTMC class to set its bulk-in EP to ACK so that it can
338// receive USBTMC commands.
339// Returns false if it was already in an ACK state or is busy
340// processing a command (such as a clear). Returns true if it was
341// in the NAK state and successfully transitioned to the ACK wait
342// state.
343bool tud_usbtmc_start_bus_read()
344{
345 usbtmcd_state_enum oldState = usbtmc_state.state;
346 switch(oldState)
347 {
348 // These may transition to IDLE
349 case STATE_NAK:
350 case STATE_ABORTING_BULK_IN_ABORTED:
351 TU_VERIFY(atomicChangeState(oldState, STATE_IDLE));
352 break;
353 // When receiving, let it remain receiving
354 case STATE_RCV:
355 break;
356 default:
357 TU_VERIFY(false);
358 }
359 TU_VERIFY(usbd_edpt_xfer(usbtmc_state.rhport, usbtmc_state.ep_bulk_out, usbtmc_state.ep_bulk_out_buf, 64));
360 return true;
361}

363void usbtmcd_reset_cb(uint8_t rhport)
364{
365 (void)rhport;
366 usbtmc_capabilities_specific_t const * capabilities = tud_usbtmc_get_capabilities_cb();
367
368 criticalEnter();

375static bool handle_devMsgOutStart(uint8_t rhport, void *data, size_t len)
376{
377 (void)rhport;
378 // return true upon failure, as we can assume error is being handled elsewhere.
379 TU_VERIFY(atomicChangeState(STATE_IDLE, STATE_RCV), true);
380 usbtmc_state.transfer_size_sent = 0u;
381
382 // must be a header, should have been confirmed before calling here.
383 usbtmc_msg_request_dev_dep_out *msg = (usbtmc_msg_request_dev_dep_out*)data;
384 usbtmc_state.transfer_size_remaining = msg->TransferSize;
385 TU_VERIFY(tud_usbtmc_msgBulkOut_start_cb(msg));
386
387 TU_VERIFY(handle_devMsgOut(rhport, (uint8_t*)data + sizeof(*msg), len - sizeof(*msg), len));
388 usbtmc_state.lastBulkOutTag = msg->header.bTag;
389 return true;
390}
391
392static bool handle_devMsgOut(uint8_t rhport, void *data, size_t len, size_t packetLen)
393{
394 (void)rhport;
395 // return true upon failure, as we can assume error is being handled elsewhere.
396 TU_VERIFY(usbtmc_state.state == STATE_RCV,true);
397
398 bool shortPacket = (packetLen < USBTMCD_MAX_PACKET_SIZE);
399
400 // Packet is to be considered complete when we get enough data or at a short packet.
401 bool atEnd = false;
402 if(len >= usbtmc_state.transfer_size_remaining || shortPacket)
403 {
404 atEnd = true;
405 TU_VERIFY(atomicChangeState(STATE_RCV, STATE_NAK));
406 }
407
408 len = tu_min32(len, usbtmc_state.transfer_size_remaining);
409
410 usbtmc_state.transfer_size_remaining -= len;
411 usbtmc_state.transfer_size_sent += len;
412
413 // App may (should?) call the wait_for_bus() command at this point
414 if(!tud_usbtmc_msg_data_cb(data, len, atEnd))
415 {
416 // TODO: Go to an error state upon failure other than just stalling the EP?
417 return false;
418 }
419
420
421 return true;
422}
423
424static bool handle_devMsgIn(void *data, size_t len)
425{
426 TU_VERIFY(len == sizeof(usbtmc_msg_request_dev_dep_in));
427 usbtmc_msg_request_dev_dep_in *msg = (usbtmc_msg_request_dev_dep_in*)data;
428 bool stateChanged = atomicChangeState(STATE_IDLE, STATE_TX_REQUESTED);
429 TU_VERIFY(stateChanged);
430 usbtmc_state.lastBulkInTag = msg->header.bTag;
431 usbtmc_state.transfer_size_remaining = msg->TransferSize;
432 usbtmc_state.transfer_size_sent = 0u;
433
434 termCharRequested = msg->bmTransferAttributes.TermCharEnabled;
435 termChar = msg->TermChar;
436
437 if(termCharRequested)
438 TU_VERIFY(usbtmc_state.capabilities->bmDevCapabilities.canEndBulkInOnTermChar);
439
440 TU_VERIFY(tud_usbtmc_msgBulkIn_request_cb(msg));
441 return true;
442}

444bool usbtmcd_xfer_cb(uint8_t rhport, uint8_t ep_addr, xfer_result_t result, uint32_t xferred_bytes)
445{
446 TU_VERIFY(result == XFER_RESULT_SUCCESS);
447 //uart_tx_str_sync("TMC XFER CB\r\n");
448 if(usbtmc_state.state == STATE_CLEARING) {
449 return true; /* I think we can ignore everything here */
450 }
451
452 if(ep_addr == usbtmc_state.ep_bulk_out)
453 {
454 usbtmc_msg_generic_t *msg = NULL;
455
456 switch(usbtmc_state.state)
457 {
458 case STATE_IDLE:
459 TU_VERIFY(xferred_bytes >= sizeof(usbtmc_msg_generic_t));
460 msg = (usbtmc_msg_generic_t*)(usbtmc_state.ep_bulk_out_buf);
461 uint8_t invInvTag = (uint8_t)~(msg->header.bTagInverse);
462 TU_VERIFY(msg->header.bTag == invInvTag);
463 TU_VERIFY(msg->header.bTag != 0x00);
464
465 switch(msg->header.MsgID) {
466 case USBTMC_MSGID_DEV_DEP_MSG_OUT:
467 if(!handle_devMsgOutStart(rhport, msg, xferred_bytes))
468 {
469 usbd_edpt_stall(rhport, usbtmc_state.ep_bulk_out);
470 TU_VERIFY(false);
471 }
472 break;
473
474 case USBTMC_MSGID_DEV_DEP_MSG_IN:
475 TU_VERIFY(handle_devMsgIn(msg, xferred_bytes));
476 break;
477
478#if (CFG_TUD_USBTMC_ENABLE_488)
479 case USBTMC_MSGID_USB488_TRIGGER:
480 // Spec says we halt the EP if we didn't declare we support it.
481 TU_VERIFY(usbtmc_state.capabilities->bmIntfcCapabilities488.supportsTrigger);
482 TU_VERIFY(tud_usbtmc_msg_trigger_cb(msg));
483
484 break;
485#endif
486 case USBTMC_MSGID_VENDOR_SPECIFIC_MSG_OUT:
487 case USBTMC_MSGID_VENDOR_SPECIFIC_IN:
488 default:
489 usbd_edpt_stall(rhport, usbtmc_state.ep_bulk_out);
490 TU_VERIFY(false);
491 return false;
492 }
493 return true;
494
495 case STATE_RCV:
496 if(!handle_devMsgOut(rhport, usbtmc_state.ep_bulk_out_buf, xferred_bytes, xferred_bytes))
497 {
498 usbd_edpt_stall(rhport, usbtmc_state.ep_bulk_out);
499 TU_VERIFY(false);
500 }
501 return true;
502
503 case STATE_ABORTING_BULK_OUT:
504 TU_VERIFY(false);
505 return false; // Should be stalled by now, shouldn't have received a packet.
506
507 case STATE_TX_REQUESTED:
508 case STATE_TX_INITIATED:
509 case STATE_ABORTING_BULK_IN:
510 case STATE_ABORTING_BULK_IN_SHORTED:
511 case STATE_ABORTING_BULK_IN_ABORTED:
512 default:
513 TU_VERIFY(false);
514 }
515 }
516 else if(ep_addr == usbtmc_state.ep_bulk_in)
517 {
518 switch(usbtmc_state.state) {
519 case STATE_TX_SHORTED:
520 TU_VERIFY(atomicChangeState(STATE_TX_SHORTED, STATE_NAK));
521 TU_VERIFY(tud_usbtmc_msgBulkIn_complete_cb());
522 break;
523
524 case STATE_TX_INITIATED:
525 if(usbtmc_state.transfer_size_remaining >=sizeof(usbtmc_state.ep_bulk_in_buf))
526 {
527 // FIXME! This removes const below!
528 TU_VERIFY( usbd_edpt_xfer(rhport, usbtmc_state.ep_bulk_in,
529 (void*)(uintptr_t) usbtmc_state.devInBuffer, sizeof(usbtmc_state.ep_bulk_in_buf)));
530 usbtmc_state.devInBuffer += sizeof(usbtmc_state.ep_bulk_in_buf);
531 usbtmc_state.transfer_size_remaining -= sizeof(usbtmc_state.ep_bulk_in_buf);
532 usbtmc_state.transfer_size_sent += sizeof(usbtmc_state.ep_bulk_in_buf);
533 }
534 else // last packet
535 {
536 size_t packetLen = usbtmc_state.transfer_size_remaining;
537 memcpy(usbtmc_state.ep_bulk_in_buf, usbtmc_state.devInBuffer, usbtmc_state.transfer_size_remaining);
538 usbtmc_state.transfer_size_sent += sizeof(usbtmc_state.transfer_size_remaining);
539 usbtmc_state.transfer_size_remaining = 0;
540 usbtmc_state.devInBuffer = NULL;
541 TU_VERIFY( usbd_edpt_xfer(rhport, usbtmc_state.ep_bulk_in, usbtmc_state.ep_bulk_in_buf, (uint16_t)packetLen) );
542 if(((packetLen % USBTMCD_MAX_PACKET_SIZE) != 0) || (packetLen == 0 ))
543 {
544 usbtmc_state.state = STATE_TX_SHORTED;
545 }
546 }
547 return true;
548
549 case STATE_ABORTING_BULK_IN:
550 // need to send short packet (ZLP?)
551 TU_VERIFY( usbd_edpt_xfer(rhport, usbtmc_state.ep_bulk_in, usbtmc_state.ep_bulk_in_buf,(uint16_t)0u));
552 usbtmc_state.state = STATE_ABORTING_BULK_IN_SHORTED;
553 return true;
554
555 case STATE_ABORTING_BULK_IN_SHORTED:
556 /* Done. :)*/
557 usbtmc_state.state = STATE_ABORTING_BULK_IN_ABORTED;
558 return true;
559
560 default:
561 TU_ASSERT(false);
562 return false;
563 }
564 }
565 else if (ep_addr == usbtmc_state.ep_int_in) {
566 // Good?
567 return true;
568 }
569 return false;
570}

572// Invoked when a control transfer occurred on an interface of this class
573// Driver response accordingly to the request and the transfer stage (setup/data/ack)
574// return false to stall control endpoint (e.g unsupported request)
575bool usbtmcd_control_xfer_cb(uint8_t rhport, uint8_t stage, tusb_control_request_t const * request)
576{
577 // nothing to do with DATA and ACK stage
578 if ( stage != CONTROL_STAGE_SETUP ) return true;
579
580 uint8_t tmcStatusCode = USBTMC_STATUS_FAILED;
581#if (CFG_TUD_USBTMC_ENABLE_488)
582 uint8_t bTag;
583#endif
584
585 if((request->bmRequestType_bit.type == TUSB_REQ_TYPE_STANDARD) &&
586 (request->bmRequestType_bit.recipient == TUSB_REQ_RCPT_ENDPOINT) &&
587 (request->bRequest == TUSB_REQ_CLEAR_FEATURE) &&
588 (request->wValue == TUSB_REQ_FEATURE_EDPT_HALT))
589 {
590 uint32_t ep_addr = (request->wIndex);
591
592 if(ep_addr == usbtmc_state.ep_bulk_out)
593 {
594 criticalEnter();
595 usbtmc_state.state = STATE_NAK; // USBD core has placed EP in NAK state for us
596 criticalLeave();
597 tud_usbtmc_bulkOut_clearFeature_cb();
598 }
599 else if (ep_addr == usbtmc_state.ep_bulk_in)
600 {
601 tud_usbtmc_bulkIn_clearFeature_cb();
602 }
603 else
604 {
605 return false;
606 }
607 return true;
608 }
609
610 // Otherwise, we only handle class requests.
611 if(request->bmRequestType_bit.type != TUSB_REQ_TYPE_CLASS)
612 {
613 return false;
614 }
615
616 // Verification that we own the interface is unneeded since it's been routed to us specifically.
617
618 switch(request->bRequest)
619 {
620 // USBTMC required requests
621 case USBTMC_bREQUEST_INITIATE_ABORT_BULK_OUT:
622 {
623 usbtmc_initiate_abort_rsp_t rsp = {
624 .bTag = usbtmc_state.lastBulkOutTag,
625 };
626 TU_VERIFY(request->bmRequestType == 0xA2); // in,class,interface
627 TU_VERIFY(request->wLength == sizeof(rsp));
628 TU_VERIFY(request->wIndex == usbtmc_state.ep_bulk_out);
629
630 // wValue is the requested bTag to abort
631 if(usbtmc_state.state != STATE_RCV)
632 {
633 rsp.USBTMC_status = USBTMC_STATUS_FAILED;
634 }
635 else if(usbtmc_state.lastBulkOutTag == (request->wValue & 0x7Fu))
636 {
637 rsp.USBTMC_status = USBTMC_STATUS_TRANSFER_NOT_IN_PROGRESS;
638 }
639 else
640 {
641 rsp.USBTMC_status = USBTMC_STATUS_SUCCESS;
642 // Check if we've queued a short packet
643 criticalEnter();
644 usbtmc_state.state = STATE_ABORTING_BULK_OUT;
645 criticalLeave();
646 TU_VERIFY(tud_usbtmc_initiate_abort_bulk_out_cb(&(rsp.USBTMC_status)));
647 usbd_edpt_stall(rhport, usbtmc_state.ep_bulk_out);
648 }
649 TU_VERIFY(tud_control_xfer(rhport, request, (void*)&rsp,sizeof(rsp)));
650 return true;
651 }
652
653 case USBTMC_bREQUEST_CHECK_ABORT_BULK_OUT_STATUS:
654 {
655 usbtmc_check_abort_bulk_rsp_t rsp = {
656 .USBTMC_status = USBTMC_STATUS_SUCCESS,
657 .NBYTES_RXD_TXD = usbtmc_state.transfer_size_sent
658 };
659 TU_VERIFY(request->bmRequestType == 0xA2); // in,class,EP
660 TU_VERIFY(request->wLength == sizeof(rsp));
661 TU_VERIFY(request->wIndex == usbtmc_state.ep_bulk_out);
662 TU_VERIFY(tud_usbtmc_check_abort_bulk_out_cb(&rsp));
663 TU_VERIFY(tud_control_xfer(rhport, request, (void*)&rsp,sizeof(rsp)));
664 return true;
665 }
666
667 case USBTMC_bREQUEST_INITIATE_ABORT_BULK_IN:
668 {
669 usbtmc_initiate_abort_rsp_t rsp = {
670 .bTag = usbtmc_state.lastBulkInTag,
671 };
672 TU_VERIFY(request->bmRequestType == 0xA2); // in,class,interface
673 TU_VERIFY(request->wLength == sizeof(rsp));
674 TU_VERIFY(request->wIndex == usbtmc_state.ep_bulk_in);
675 // wValue is the requested bTag to abort
676 if((usbtmc_state.state == STATE_TX_REQUESTED || usbtmc_state.state == STATE_TX_INITIATED) &&
677 usbtmc_state.lastBulkInTag == (request->wValue & 0x7Fu))
678 {
679 rsp.USBTMC_status = USBTMC_STATUS_SUCCESS;
680 usbtmc_state.transfer_size_remaining = 0u;
681 // Check if we've queued a short packet
682 criticalEnter();
683 usbtmc_state.state = ((usbtmc_state.transfer_size_sent % USBTMCD_MAX_PACKET_SIZE) == 0) ?
684 STATE_ABORTING_BULK_IN : STATE_ABORTING_BULK_IN_SHORTED;
685 criticalLeave();
686 if(usbtmc_state.transfer_size_sent == 0)
687 {
688 // Send short packet, nothing is in the buffer yet
689 TU_VERIFY( usbd_edpt_xfer(rhport, usbtmc_state.ep_bulk_in, usbtmc_state.ep_bulk_in_buf,(uint16_t)0u));
690 usbtmc_state.state = STATE_ABORTING_BULK_IN_SHORTED;
691 }
692 TU_VERIFY(tud_usbtmc_initiate_abort_bulk_in_cb(&(rsp.USBTMC_status)));
693 }
694 else if((usbtmc_state.state == STATE_TX_REQUESTED || usbtmc_state.state == STATE_TX_INITIATED))
695 { // FIXME: Unsure how to check if the OUT endpoint fifo is non-empty....
696 rsp.USBTMC_status = USBTMC_STATUS_TRANSFER_NOT_IN_PROGRESS;
697 }
698 else
699 {
700 rsp.USBTMC_status = USBTMC_STATUS_FAILED;
701 }
702 TU_VERIFY(tud_control_xfer(rhport, request, (void*)&rsp,sizeof(rsp)));
703 return true;
704 }
705
706 case USBTMC_bREQUEST_CHECK_ABORT_BULK_IN_STATUS:
707 {
708 TU_VERIFY(request->bmRequestType == 0xA2); // in,class,EP
709 TU_VERIFY(request->wLength == 8u);
710
711 usbtmc_check_abort_bulk_rsp_t rsp =
712 {
713 .USBTMC_status = USBTMC_STATUS_FAILED,
714 .bmAbortBulkIn =
715 {
716 .BulkInFifoBytes = (usbtmc_state.state != STATE_ABORTING_BULK_IN_ABORTED)
717 },
718 .NBYTES_RXD_TXD = usbtmc_state.transfer_size_sent,
719 };
720 TU_VERIFY(tud_usbtmc_check_abort_bulk_in_cb(&rsp));
721 criticalEnter();
722 switch(usbtmc_state.state)
723 {
724 case STATE_ABORTING_BULK_IN_ABORTED:
725 rsp.USBTMC_status = USBTMC_STATUS_SUCCESS;
726 usbtmc_state.state = STATE_IDLE;
727 break;
728 case STATE_ABORTING_BULK_IN:
729 case STATE_ABORTING_BULK_OUT:
730 rsp.USBTMC_status = USBTMC_STATUS_PENDING;
731 break;
732 default:
733 break;
734 }
735 criticalLeave();
736 TU_VERIFY(tud_control_xfer(rhport, request, (void*)&rsp,sizeof(rsp)));
737
738 return true;
739 }
740
741 case USBTMC_bREQUEST_INITIATE_CLEAR:
742 {
743 TU_VERIFY(request->bmRequestType == 0xA1); // in,class,interface
744 TU_VERIFY(request->wLength == sizeof(tmcStatusCode));
745 // After receiving an INITIATE_CLEAR request, the device must Halt the Bulk-OUT endpoint, queue the
746 // control endpoint response shown in Table 31, and clear all input buffers and output buffers.
747 usbd_edpt_stall(rhport, usbtmc_state.ep_bulk_out);
748 usbtmc_state.transfer_size_remaining = 0;
749 criticalEnter();
750 usbtmc_state.state = STATE_CLEARING;
751 criticalLeave();
752 TU_VERIFY(tud_usbtmc_initiate_clear_cb(&tmcStatusCode));
753 TU_VERIFY(tud_control_xfer(rhport, request, (void*)&tmcStatusCode,sizeof(tmcStatusCode)));
754 return true;
755 }
756
757 case USBTMC_bREQUEST_CHECK_CLEAR_STATUS:
758 {
759 TU_VERIFY(request->bmRequestType == 0xA1); // in,class,interface
760 usbtmc_get_clear_status_rsp_t clearStatusRsp = {0};
761 TU_VERIFY(request->wLength == sizeof(clearStatusRsp));
762
763 if(usbd_edpt_busy(rhport, usbtmc_state.ep_bulk_in))
764 {
765 // Stuff stuck in TX buffer?
766 clearStatusRsp.bmClear.BulkInFifoBytes = 1;
767 clearStatusRsp.USBTMC_status = USBTMC_STATUS_PENDING;
768 }
769 else
770 {
771 // Let app check if it's clear
772 TU_VERIFY(tud_usbtmc_check_clear_cb(&clearStatusRsp));
773 }
774 if(clearStatusRsp.USBTMC_status == USBTMC_STATUS_SUCCESS)
775 {
776 criticalEnter();
777 usbtmc_state.state = STATE_IDLE;
778 criticalLeave();
779 }
780 TU_VERIFY(tud_control_xfer(rhport, request, (void*)&clearStatusRsp,sizeof(clearStatusRsp)));
781 return true;
782 }
783
784 case USBTMC_bREQUEST_GET_CAPABILITIES:
785 {
786 TU_VERIFY(request->bmRequestType == 0xA1); // in,class,interface
787 TU_VERIFY(request->wLength == sizeof(*(usbtmc_state.capabilities)));
788 TU_VERIFY(tud_control_xfer(rhport, request, (void*)(uintptr_t) usbtmc_state.capabilities, sizeof(*usbtmc_state.capabilities)));
789 return true;
790 }
791 // USBTMC Optional Requests
792
793 case USBTMC_bREQUEST_INDICATOR_PULSE: // Optional
794 {
795 TU_VERIFY(request->bmRequestType == 0xA1); // in,class,interface
796 TU_VERIFY(request->wLength == sizeof(tmcStatusCode));
797 TU_VERIFY(usbtmc_state.capabilities->bmIntfcCapabilities.supportsIndicatorPulse);
798 TU_VERIFY(tud_usbtmc_indicator_pulse_cb(request, &tmcStatusCode));
799 TU_VERIFY(tud_control_xfer(rhport, request, (void*)&tmcStatusCode, sizeof(tmcStatusCode)));
800 return true;
801 }
802#if (CFG_TUD_USBTMC_ENABLE_488)
803
804 // USB488 required requests
805 case USB488_bREQUEST_READ_STATUS_BYTE:
806 {
807 usbtmc_read_stb_rsp_488_t rsp;
808 TU_VERIFY(request->bmRequestType == 0xA1); // in,class,interface
809 TU_VERIFY(request->wLength == sizeof(rsp)); // in,class,interface
810
811 bTag = request->wValue & 0x7F;
812 TU_VERIFY(request->bmRequestType == 0xA1);
813 TU_VERIFY((request->wValue & (~0x7F)) == 0u); // Other bits are required to be zero
814 TU_VERIFY(bTag >= 0x02 && bTag <= 127);
815 TU_VERIFY(request->wIndex == usbtmc_state.itf_id);
816 TU_VERIFY(request->wLength == 0x0003);
817 rsp.bTag = (uint8_t)bTag;
818 if(usbtmc_state.ep_int_in != 0)
819 {
820 rsp.USBTMC_status = USBTMC_STATUS_SUCCESS;
821 rsp.statusByte = 0x00; // Use interrupt endpoint, instead.
822
823 usbtmc_read_stb_interrupt_488_t intMsg =
824 {
825 .bNotify1 = {
826 .one = 1,
827 .bTag = bTag & 0x7Fu,
828 },
829 .StatusByte = tud_usbtmc_get_stb_cb(&(rsp.USBTMC_status))
830 };
831 usbd_edpt_xfer(rhport, usbtmc_state.ep_int_in, (void*)&intMsg, sizeof(intMsg));
832 }
833 else
834 {
835 rsp.statusByte = tud_usbtmc_get_stb_cb(&(rsp.USBTMC_status));
836 }
837 TU_VERIFY(tud_control_xfer(rhport, request, (void*)&rsp, sizeof(rsp)));
838 return true;
839 }
840 // USB488 optional requests
841 case USB488_bREQUEST_REN_CONTROL:
842 case USB488_bREQUEST_GO_TO_LOCAL:
843 case USB488_bREQUEST_LOCAL_LOCKOUT:
844 {
845 TU_VERIFY(request->bmRequestType == 0xA1); // in,class,interface
846 TU_VERIFY(false);
847 return false;
848 }
849#endif
850
851 default:
852 TU_VERIFY(false);
853 return false;
854 }
855 TU_VERIFY(false);
856}
