23void MidiHandler::midiRx(uint8_t itf,uint8_t packet[4]){
24
25 pulseSysLed();

27 uint8_t packettype = packet[1] & 0xf0;
28 uint8_t b1 = packet[2];
29 uint8_t b2 = packet[3];
30
31 switch (packettype) {
32 case 0x80:
33 // Noteoff

36 case 0x90:
37 // Noteon

43 break;

47 break;
48 case 0xC0:
49 //pc
50 break;

52

54 case 0xE0:

56 //pb

60 }

63 }
