193static inline void pcd_set_ep_cnt_rx_reg(__O uint16_t * pdwReg, size_t wCount) {
194 uint32_t wNBlocks;
195 if(wCount > 62u)
196 {
197 wNBlocks = wCount >> 5u;
198 if((wCount & 0x1fU) == 0u)
199 {
200 wNBlocks--;
201 }
202 wNBlocks = wNBlocks << 10u;
203 wNBlocks |= 0x8000u; // Mark block size as 32byte
204 *pdwReg = (uint16_t)wNBlocks;
205 }
206 else
207 {
208 wNBlocks = wCount >> 1u;
209 if((wCount & 0x1U) != 0u)
210 {
211 wNBlocks++;
212 }
213 *pdwReg = (uint16_t)((wNBlocks) << 10u);
214 }
215}

279static inline void pcd_set_ep_tx_status(USB_TypeDef * USBx, uint32_t bEpNum, uint32_t wState)
280{
281 uint32_t regVal = pcd_get_endpoint(USBx, bEpNum);
282 regVal &= USB_EPTX_DTOGMASK;
283
284 /* toggle first bit ? */
285 if((USB_EPTX_DTOG1 & (wState))!= 0U)
286 {
287 regVal ^= USB_EPTX_DTOG1;
288 }
289 /* toggle second bit ? */
290 if((USB_EPTX_DTOG2 & ((uint32_t)(wState)))!= 0U)
291 {
292 regVal ^= USB_EPTX_DTOG2;
293 }
294 regVal |= USB_EP_CTR_RX|USB_EP_CTR_TX;
295 pcd_set_endpoint(USBx, bEpNum, regVal);
296} /* pcd_set_ep_tx_status */
