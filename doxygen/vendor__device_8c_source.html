82//--------------------------------------------------------------------+
83// Read API
84//--------------------------------------------------------------------+
85static void _prep_out_transaction (vendord_interface_t* p_itf)
86{
87 // skip if previous transfer not complete
88 if ( usbd_edpt_busy(TUD_OPT_RHPORT, p_itf->ep_out) ) return;
89
90 // Prepare for incoming data but only allow what we can store in the ring buffer.
91 uint16_t max_read = tu_fifo_remaining(&p_itf->rx_ff);
92 if ( max_read >= CFG_TUD_VENDOR_EPSIZE )
93 {
94 usbd_edpt_xfer(TUD_OPT_RHPORT, p_itf->ep_out, p_itf->epout_buf, CFG_TUD_VENDOR_EPSIZE);
95 }
96}

98uint32_t tud_vendor_n_read (uint8_t itf, void* buffer, uint32_t bufsize)
99{
100 vendord_interface_t* p_itf = &_vendord_itf[itf];
101 uint32_t num_read = tu_fifo_read_n(&p_itf->rx_ff, buffer, bufsize);
102 _prep_out_transaction(p_itf);
103 return num_read;
104}

109 tu_fifo_clear(&p_itf->rx_ff);
110 _prep_out_transaction(p_itf);

113//--------------------------------------------------------------------+
114// Write API
115//--------------------------------------------------------------------+
116static bool maybe_transmit(vendord_interface_t* p_itf)
117{
118 // skip if previous transfer not complete
119 TU_VERIFY( !usbd_edpt_busy(TUD_OPT_RHPORT, p_itf->ep_in) );
120
121 uint16_t count = tu_fifo_read_n(&p_itf->tx_ff, p_itf->epin_buf, CFG_TUD_VENDOR_EPSIZE);
122 if (count > 0)
123 {
124 TU_ASSERT( usbd_edpt_xfer(TUD_OPT_RHPORT, p_itf->ep_in, p_itf->epin_buf, count) );
125 }
126 return true;
127}

145void vendord_init(void)
146{
147 tu_memclr(_vendord_itf, sizeof(_vendord_itf));
148
149 for(uint8_t i=0; i<CFG_TUD_VENDOR; i++)
150 {
151 vendord_interface_t* p_itf = &_vendord_itf[i];
152
153 // config fifo
154 tu_fifo_config(&p_itf->rx_ff, p_itf->rx_ff_buf, CFG_TUD_VENDOR_RX_BUFSIZE, 1, false);
155 tu_fifo_config(&p_itf->tx_ff, p_itf->tx_ff_buf, CFG_TUD_VENDOR_TX_BUFSIZE, 1, false);
156
157#if CFG_FIFO_MUTEX
158 tu_fifo_config_mutex(&p_itf->rx_ff, NULL, osal_mutex_create(&p_itf->rx_ff_mutex));
159 tu_fifo_config_mutex(&p_itf->tx_ff, osal_mutex_create(&p_itf->tx_ff_mutex), NULL);
160#endif
161 }
162}

178uint16_t vendord_open(uint8_t rhport, tusb_desc_interface_t const * desc_itf, uint16_t max_len)
179{
180 TU_VERIFY(TUSB_CLASS_VENDOR_SPECIFIC == desc_itf->bInterfaceClass, 0);
181
182 uint8_t const * p_desc = tu_desc_next(desc_itf);
183 uint8_t const * desc_end = p_desc + max_len;
184
185 // Find available interface
186 vendord_interface_t* p_vendor = NULL;
187 for(uint8_t i=0; i<CFG_TUD_VENDOR; i++)
188 {
189 if ( _vendord_itf[i].ep_in == 0 && _vendord_itf[i].ep_out == 0 )
190 {
191 p_vendor = &_vendord_itf[i];
192 break;
193 }
194 }
195 TU_VERIFY(p_vendor, 0);
196
197 p_vendor->itf_num = desc_itf->bInterfaceNumber;
198 if (desc_itf->bNumEndpoints)
199 {
200 // skip non-endpoint descriptors
201 while ( (TUSB_DESC_ENDPOINT != tu_desc_type(p_desc)) && (p_desc < desc_end) )
202 {
203 p_desc = tu_desc_next(p_desc);
204 }
205
206 // Open endpoint pair with usbd helper
207 TU_ASSERT(usbd_open_edpt_pair(rhport, p_desc, desc_itf->bNumEndpoints, TUSB_XFER_BULK, &p_vendor->ep_out, &p_vendor->ep_in), 0);
208
209 p_desc += desc_itf->bNumEndpoints*sizeof(tusb_desc_endpoint_t);
210
211 // Prepare for incoming data
212 if ( p_vendor->ep_out )
213 {
214 TU_ASSERT(usbd_edpt_xfer(rhport, p_vendor->ep_out, p_vendor->epout_buf, sizeof(p_vendor->epout_buf)), 0);
215 }
216
217 if ( p_vendor->ep_in ) maybe_transmit(p_vendor);
218 }
219
220 return (uintptr_t) p_desc - (uintptr_t) desc_itf;
221}

223bool vendord_xfer_cb(uint8_t rhport, uint8_t ep_addr, xfer_result_t result, uint32_t xferred_bytes)
224{
225 (void) rhport;
226 (void) result;
227
228 uint8_t itf = 0;
229 vendord_interface_t* p_itf = _vendord_itf;
230
231 for ( ; ; itf++, p_itf++)
232 {
233 if (itf >= TU_ARRAY_SIZE(_vendord_itf)) return false;
234
235 if ( ( ep_addr == p_itf->ep_out ) || ( ep_addr == p_itf->ep_in ) ) break;
236 }
237
238 if ( ep_addr == p_itf->ep_out )
239 {
240 // Receive new data
241 tu_fifo_write_n(&p_itf->rx_ff, p_itf->epout_buf, xferred_bytes);
242
243 // Invoked callback if any
244 if (tud_vendor_rx_cb) tud_vendor_rx_cb(itf);
245
246 _prep_out_transaction(p_itf);
247 }
248 else if ( ep_addr == p_itf->ep_in )
249 {
250 // Send complete, try to send more if possible
251 maybe_transmit(p_itf);
252 }
253
254 return true;
255}
