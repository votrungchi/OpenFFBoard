97void FFBHIDMain::restoreFlash(){
98
99 Flash_Read(ADR_FFBWHEEL_BUTTONCONF, &this->btnsources);
100 setBtnTypes(this->btnsources);
101
102 Flash_Read(ADR_FFBWHEEL_ANALOGCONF, &this->ainsources);

107 uint8_t rateidx = conf1 & 0x3;

130void FFBHIDMain::update(){
131 if(control.request_update_disabled) {
132 //logSerial("request update disabled");
133 control.update_disabled = true;
134 control.request_update_disabled = false;
135
136 }
137 if(control.update_disabled){
138 //logSerial("Update disabled");
139 return;
140 }
141 if(control.emergency){
142 pulseErrLed();
143 return;
144 }
145 if(control.resetEncoder){
146 control.resetEncoder = false;
147 axes_manager->resetPosZero();
148 }
149
150
151
152 // TODO Emulate a SOF timer...
153 if(HAL_GetTick() - lastUsbReportTick > 0 && !control.usb_disabled){
154 lastUsbReportTick = HAL_GetTick();
155 control.usb_update_flag = true;
156 }
157
158 // If either usb or timer triggered
159 if(control.usb_update_flag || control.update_flag){
160 axes_manager->update();
161 control.update_flag = false;
162 if(control.usb_update_flag){
163 control.usb_update_flag = false;
164 if(++report_rate_cnt >= usb_report_rate){
165 report_rate_cnt = 0;
166 this->send_report();
167 }
168 }
169 axes_manager->updateTorque();
170 }
171}

250void FFBHIDMain::send_report(){
251 if(!sourcesSem.Take(10)){
252 return;
253 }
254 // Read buttons
255 reportHID.buttons = 0; // Reset buttons
256 uint8_t shift = 0;
257 if(btns.size() != 0){
258 for(auto &btn : btns){
259 uint64_t buf = 0;
260 uint8_t amount = btn->readButtons(&buf);
261 reportHID.buttons |= buf << shift;
262 shift += amount;
263 }
264 }
265
266
267 // Encoder
268 //axes_manager->addAxesToReport(analogAxesReport, &count);
269
270 std::vector<int32_t>* axes = axes_manager->getAxisValues();
271 uint8_t count = 0;
272 for(auto val : *axes){
273 setHidReportAxis(&reportHID,count++,val);
274 }
275
276 // Fill remaining values with analog inputs
277 for(auto &ain : analog_inputs){
278 std::vector<int32_t>* axes = ain->getAxes();
279 for(int32_t val : *axes){
280 if(count >= analogAxisCount)
281 break;
282 setHidReportAxis(&reportHID,count++,val);
283 }
284 }
285 sourcesSem.Give();
286 // Fill rest
287 for(;count<analogAxisCount; count++){
288 setHidReportAxis(&reportHID,count,0);
289 }
290
291
292 /*
293 * Only send a new report if actually changed since last time or timeout and hid is ready
294 */
295 if( (reportSendCounter++ > 100/usb_report_rate || (memcmp(&lastReportHID,&reportHID,sizeof(reportHID_t)) != 0) )
296 && tud_hid_n_ready(0)
297 && !(reportSendCounter < usb_report_rate*2 && this->hidCommands->waitingToSend())) // Check if HID command interface wants to send something and allow that if we did not skip the last report
298 {
299
300
301 tud_hid_report(0, reinterpret_cast<uint8_t*>(&reportHID), sizeof(reportHID_t));
302 lastReportHID = reportHID;
303 reportSendCounter = 0;
304
305 }
