31void CanAnalogBase::setupCanPort(){
32 if(filterId != -1){
33 this->port->removeCanFilter(filterId);
34 }
35 CAN_FilterTypeDef sFilterConfig;
36 sFilterConfig.FilterBank = 0;
37 sFilterConfig.FilterMode = CAN_FILTERMODE_IDMASK;
38 sFilterConfig.FilterScale = CAN_FILTERSCALE_32BIT;
39 sFilterConfig.FilterIdHigh = (canId << 5);
40 sFilterConfig.FilterIdLow = 0x0000;
41 sFilterConfig.FilterMaskIdHigh = ~((canId << 5) ^ ((canId+1) << 5)); // Mask is inverted xor of the IDs
42 sFilterConfig.FilterMaskIdLow = 0x0000;
43 sFilterConfig.FilterFIFOAssignment = CAN_RX_FIFO0;
44 sFilterConfig.FilterActivation = ENABLE;
45 sFilterConfig.SlaveStartFilterBank = 14;
46 this->filterId = this->port->addCanFilter(sFilterConfig);
47
48 //this->port->setSpeedPreset(CANSPEEDPRESET_500); // default speed used
49 this->port->start();

84CommandStatus CanAnalogBase::command(const ParsedCommand& cmd,std::vector<CommandReply>& replies){
85
86 switch(static_cast<CanAnalog_commands>(cmd.cmdId)){
87
88 case CanAnalog_commands::amount:
89 if(cmd.type == CMDtype::get){
90 replies.emplace_back(this->axes);
91 }else if(cmd.type == CMDtype::set){
92 setAxisNum(cmd.val);
93 }
94 break;
95
96 case CanAnalog_commands::maxAmount:
97 replies.emplace_back(this->maxAxes);
98 break;
99
100 case CanAnalog_commands::canid:
101 if(cmd.type == CMDtype::get){
102 replies.emplace_back(this->canId);
103 }else if(cmd.type == CMDtype::set){
104 canId = std::min<uint32_t>((cmd.val) & 0x7ff,0x7ff-(this->axes/4));
105 setupCanPort(); // Set can filter
106 }
107 break;
108
109 default:
110 return CommandStatus::NOT_FOUND;
111 }
112 return CommandStatus::OK;
113}

115void CanAnalogBase::canRxPendCallback(CAN_HandleTypeDef *hcan,uint8_t* rxBuf,CAN_RxHeaderTypeDef* rxHeader,uint32_t fifo){
116
117 uint32_t id = (rxHeader->StdId) & 0x7FF;
118 pulseClipLed();
119 if(rxHeader->RTR != CAN_RTR_DATA || rxHeader->DLC != 8){
120 return;
121 }
122
123 for(uint8_t packet = 0; packet < packets ; packet++){
124 if(id != this->canId+packet){
125 continue;
126 }
127 for(uint8_t i = 0; i< axes-(packet*4) ; i++){
128 this->buf[i+(packet*4)] = rxBuf[i*2] | (rxBuf[i*2+1] << 8);
129 }
130 }
131}
