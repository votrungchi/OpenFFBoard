156void Axis::restoreFlash(){
157 //NormalizedAxis::restoreFlash();
158 // read all constants
159 uint16_t value;
160 if (Flash_Read(flashAddrs.config, &value)){
161 this->conf = Axis::decodeConfFromInt(value);
162 }else{
163 pulseErrLed();
164 }
165
166 setDrvType(this->conf.drvtype);
167 setEncType(this->conf.enctype);
168
169 if (Flash_Read(flashAddrs.maxSpeed, &value)){
170 this->maxSpeedDegS = value;
171 }else{
172 pulseErrLed();
173 }
174//
175// if (Flash_Read(flashAddrs.maxAccel, &value)){
176// this->maxTorqueRateMS = value;
177// }else{
178// pulseErrLed();
179// }
180
181
182 uint16_t esval, power;
183 if(Flash_Read(flashAddrs.endstop, &esval)) {
184 fx_ratio_i = esval & 0xff;
185 endstopStrength = (esval >> 8) & 0xff;
186 }
187
188
189 if(Flash_Read(flashAddrs.power, &power)){
190 setPower(power);
191 }
192 uint16_t deg_t;
193 if(Flash_Read(flashAddrs.degrees, &deg_t)){
194 this->degreesOfRotation = deg_t & 0x7fff;
195 this->invertAxis = (deg_t >> 15) & 0x1;
196 setDegrees(degreesOfRotation);
197 }
198
199
200 uint16_t effects;
201 if(Flash_Read(flashAddrs.effects1, &effects)){
202 setIdleSpringStrength(effects & 0xff);
203 setDamperStrength((effects >> 8) & 0xff);
204 }
205
206 uint16_t ratio;
207 if(Flash_Read(flashAddrs.encoderRatio, &ratio)){
208 setGearRatio(ratio & 0xff, (ratio >> 8) & 0xff);
209 }
210
211 uint16_t filterStorage;
212 if (Flash_Read(flashAddrs.speedAccelFilter, &filterStorage))
213 {
214 uint8_t profile = filterStorage & 0xFF;
215 this->filterProfileId = profile;
216 speedFilter.setFc(filterSpeedCst[this->filterProfileId].freq / filter_f);
217 speedFilter.setQ(filterSpeedCst[this->filterProfileId].q / 100.0);
218 accelFilter.setFc(filterAccelCst[this->filterProfileId].freq / filter_f);
219 accelFilter.setQ(filterAccelCst[this->filterProfileId].q / 100.0);
220 }
221
222}
223// Saves parameters to flash.
224void Axis::saveFlash(){
225 //NormalizedAxis::saveFlash();
226 Flash_Write(flashAddrs.config, Axis::encodeConfToInt(this->conf));
227 Flash_Write(flashAddrs.maxSpeed, this->maxSpeedDegS);
228// Flash_Write(flashAddrs.maxAccel, (uint16_t)(this->maxTorqueRateMS));
229
230 Flash_Write(flashAddrs.endstop, fx_ratio_i | (endstopStrength << 8));
231 Flash_Write(flashAddrs.power, power);
232 Flash_Write(flashAddrs.degrees, (degreesOfRotation & 0x7fff) | (invertAxis << 15));
233 Flash_Write(flashAddrs.effects1, idlespringstrength | (damperIntensity << 8));
234 Flash_Write(flashAddrs.encoderRatio, gearRatio.numerator | (gearRatio.denominator << 8));
235
236 // save CF biquad
237 uint16_t filterStorage = (uint16_t)this->filterProfileId & 0xFF;
238 Flash_Write(flashAddrs.speedAccelFilter, filterStorage);
239}

272void Axis::prepareForUpdate(){
273 if (drv == nullptr){
274 pulseErrLed();
275 return;
276 }
277
278 //if (!drv->motorReady()) return;
279
280 float angle = getEncAngle(this->drv->getEncoder());
281
282 // Scale encoder value to set rotation range
283 // Update a change of range only when new range is within valid range
284 // if degree change, compute the SpeedScaler, it depends on degreesOfRotation
285 if (nextDegreesOfRotation != degreesOfRotation){
286 int32_t scaledEnc;
287 std::tie(scaledEnc,std::ignore) = scaleEncValue(angle, nextDegreesOfRotation);
288 if(abs(scaledEnc) < 0x7fff){
289 degreesOfRotation = nextDegreesOfRotation;
290 }
291
292 }
293
294
295 // scaledEnc now gets inverted if necessary in updateMetrics
296 int32_t scaledEnc;
297 std::tie(scaledEnc,std::ignore) = scaleEncValue(angle, degreesOfRotation);
298
299 if (abs(scaledEnc) > 0xffff && drv->motorReady()){
300 // We are way off. Shut down
301 drv->stopMotor();
302 pulseErrLed();
303 if(!outOfBounds){
304 outOfBounds = true;
305 ErrorHandler::addError(outOfBoundsError);
306 }
307
308 }else if(abs(scaledEnc) <= 0x7fff) {
309 outOfBounds = false;
310 //ErrorHandler::clearError(outOfBoundsError);
311 }
312
313 this->updateMetrics(angle);
314
315}

450std::pair<int32_t,float> Axis::scaleEncValue(float angle, uint16_t degrees){
451 if (degrees == 0){
452 return std::make_pair<int32_t,float>(0x7fff,0.0);
453 }
454
455 int32_t val = (0xffff / (float)degrees) * angle;
456 float val_f = (2.0 / (float)degrees) * angle;
457
458 return std::make_pair(val,val_f);

579void Axis::resetMetrics(float new_pos= 0) { // pos is degrees
580 metric.current = metric_t();
581 metric.current.posDegrees = new_pos;
582 std::tie(metric.current.pos,metric.current.pos_f) = scaleEncValue(new_pos, degreesOfRotation);
583 metric.previous = metric_t();

585 speedFilter.calcBiquad();

592void Axis::updateMetrics(float new_pos) { // pos is degrees
593 // store old value for next metric's computing
594 metric.previous = metric.current;
595
596 metric.current.posDegrees = new_pos;
597 std::tie(metric.current.pos,metric.current.pos_f) = scaleEncValue(new_pos, degreesOfRotation);
598
599
600 // compute speed and accel from raw instant speed normalized
601 float currentSpeed = (new_pos - metric.previous.posDegrees) * 1000.0; // deg/s
602 metric.current.speed = speedFilter.process(currentSpeed);
603 metric.current.accel = accelFilter.process((currentSpeed - _lastSpeed))* 1000.0; // deg/s/s
604 _lastSpeed = currentSpeed;
605
606 metric.current.torque = 0;
607}

634int16_t Axis::updateEndstop(){
635 int8_t clipdir = cliptest<int32_t,int32_t>(metric.current.pos, -0x7fff, 0x7fff);
636 if(clipdir == 0){
637 return 0;
638 }
639 float addtorque = clipdir*metric.current.posDegrees - (float)this->degreesOfRotation/2.0; // degress of rotation counts total range so multiply by 2
640 addtorque *= (float)endstopStrength * endstopGain; // Apply endstop gain for stiffness.
641 addtorque *= -clipdir;
642
643 return clip<int32_t,int32_t>(addtorque,-0x7fff,0x7fff);
644}

654bool Axis::updateTorque(int32_t* totalTorque) {
655
656 if(abs(effectTorque) >= 0x7fff){
657 pulseClipLed();
658 }
659
660 // Scale effect torque
661 int32_t torque = effectTorque; // Game effects
662 torque *= effect_margin_scaler;
663 torque += axisEffectTorque; // Independent effects
664 torque += updateEndstop();
665 torque *= torqueScaler; // Scale to power
666
667 // TODO speed and accel limiters
668 if(maxSpeedDegS > 0){
669
670 float torqueSign = torque > 0 ? 1 : -1; // Used to prevent metrics against the force to go into the limiter
671 // Speed. Mostly tuned...
672 spdlimiterAvg.addValue(metric.current.speed);
673 float speedreducer = (float)((spdlimiterAvg.getAverage()*torqueSign) - (float)maxSpeedDegS) * ((float)0x7FFF / maxSpeedDegS);
674 spdlimitreducerI = clip<float,int32_t>( spdlimitreducerI + ((speedreducer * 0.015) * torqueScaler),0,power);
675
676 // Accel limit. Not really useful. Maybe replace with torque slew rate limit?
677// float accreducer = (float)((metric.current.accel*torqueSign) - (float)maxAccelDegSS) * getAccelScalerNormalized();
678// acclimitreducerI = clip<float,int32_t>( acclimitreducerI + ((accreducer * 0.02) * torqueScaler),0,power);
679
680
681 // Only reduce torque. Don't invert it to prevent oscillation
682 float torqueReduction = speedreducer * speedLimiterP + spdlimitreducerI;// accreducer * 0.025 + acclimitreducerI
683 if(torque > 0){
684 torqueReduction = clip<float,int32_t>(torqueReduction,0,torque);
685 }else{
686 torqueReduction = clip<float,int32_t>(-torqueReduction,torque,0);
687 }
688
689 torque -= torqueReduction;
690 }
691 // Torque slew rate limiter
692 if(maxTorqueRateMS > 0){
693 torque = clip<int32_t,int32_t>(torque, metric.previous.torque - maxTorqueRateMS,metric.previous.torque + maxTorqueRateMS);
694 }
695// if(torque - metric.previous.torque)
696 if(outOfBounds){
697 torque = 0;
698 }
699
700 // Torque calculated. Now sending to driver
701 torque = (invertAxis) ? -torque : torque;
702 metric.current.torque = torque;
703 torque = clip<int32_t, int32_t>(torque, -power, power);
704
705 bool torqueChanged = metric.current.torque != metric.previous.torque;
706
707 if (abs(torque) == power){
708 pulseClipLed();
709 }
710
711 *totalTorque = torque;
712 return (torqueChanged);
713}

730CommandStatus Axis::command(const ParsedCommand& cmd,std::vector<CommandReply>& replies){
731
732 switch(static_cast<Axis_commands>(cmd.cmdId)){
733
734 case Axis_commands::power:
735 if (cmd.type == CMDtype::get)
736 {
737 replies.emplace_back(this->power);
738 }
739 else if (cmd.type == CMDtype::set)
740 {
741 setPower(cmd.val);
742 }
743 break;
744
745 case Axis_commands::degrees:
746 handleGetSetFunc(cmd, replies, degreesOfRotation, &Axis::setDegrees,this);
747// if (cmd.type == CMDtype::get)
748// {
749// replies.emplace_back(degreesOfRotation);
750// }
751// else if (cmd.type == CMDtype::set)
752// {
753// setDegrees(cmd.val);
754// }
755 break;
756
757 case Axis_commands::esgain:
758 handleGetSet(cmd, replies, this->endstopStrength);
759 break;
760
761 case Axis_commands::zeroenc:
762 this->setPos(0);
763 break;
764
765 case Axis_commands::invert:
766 if (cmd.type == CMDtype::get)
767 {
