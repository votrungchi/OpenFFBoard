54static inline osal_semaphore_t osal_semaphore_create(osal_semaphore_def_t* semdef)
55{
56 return xSemaphoreCreateBinaryStatic(semdef);
57}

59static inline bool osal_semaphore_post(osal_semaphore_t sem_hdl, bool in_isr)
60{
61 if ( !in_isr )
62 {
63 return xSemaphoreGive(sem_hdl) != 0;
64 }
65 else
66 {
67 BaseType_t xHigherPriorityTaskWoken;
68 BaseType_t res = xSemaphoreGiveFromISR(sem_hdl, &xHigherPriorityTaskWoken);
69
70#if CFG_TUSB_MCU == OPT_MCU_ESP32S2 || CFG_TUSB_MCU == OPT_MCU_ESP32S3
71 // not needed after https://github.com/espressif/esp-idf/commit/c5fd79547ac9b7bae06fa660e9f814d18d3390b7
72 if ( xHigherPriorityTaskWoken ) portYIELD_FROM_ISR();
73#else
74 portYIELD_FROM_ISR(xHigherPriorityTaskWoken);
75#endif
76
77 return res != 0;
78 }
79}

81static inline bool osal_semaphore_wait (osal_semaphore_t sem_hdl, uint32_t msec)
82{
83 uint32_t const ticks = (msec == OSAL_TIMEOUT_WAIT_FOREVER) ? portMAX_DELAY : pdMS_TO_TICKS(msec);
84 return xSemaphoreTake(sem_hdl, ticks);
85}

87static inline void osal_semaphore_reset(osal_semaphore_t const sem_hdl)
88{
89 xQueueReset(sem_hdl);
90}

117// role device/host is used by OS NONE for mutex (disable usb isr) only
118#define OSAL_QUEUE_DEF(_role, _name, _depth, _type) \
119 static _type _name##_##buf[_depth];\
120 osal_queue_def_t _name = { .depth = _depth, .item_sz = sizeof(_type), .buf = _name##_##buf };

122typedef struct

125 uint16_t item_sz;
126 void* buf;
127
128 StaticQueue_t sq;

133static inline osal_queue_t osal_queue_create(osal_queue_def_t* qdef)
134{
135 return xQueueCreateStatic(qdef->depth, qdef->item_sz, (uint8_t*) qdef->buf, &qdef->sq);

138static inline bool osal_queue_receive(osal_queue_t qhdl, void* data)
139{

143static inline bool osal_queue_send(osal_queue_t qhdl, void const * data, bool in_isr)
144{
145 if ( !in_isr )
146 {
147 return xQueueSendToBack(qhdl, data, OSAL_TIMEOUT_WAIT_FOREVER) != 0;
148 }
149 else
150 {
151 BaseType_t xHigherPriorityTaskWoken;
152 BaseType_t res = xQueueSendToBackFromISR(qhdl, data, &xHigherPriorityTaskWoken);
153
154#if CFG_TUSB_MCU == OPT_MCU_ESP32S2 || CFG_TUSB_MCU == OPT_MCU_ESP32S3
155 // not needed after https://github.com/espressif/esp-idf/commit/c5fd79547ac9b7bae06fa660e9f814d18d3390b7
156 if ( xHigherPriorityTaskWoken ) portYIELD_FROM_ISR();
157#else
158 portYIELD_FROM_ISR(xHigherPriorityTaskWoken);
159#endif
160
161 return res != 0;
162 }
163}
