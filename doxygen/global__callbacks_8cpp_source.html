59void HAL_ADC_ConvCpltCallback(ADC_HandleTypeDef* hadc){
60 //Pulse braking mosfet if internal voltage is higher than supply.
61 if(hadc == &VSENSE_HADC)
62 brakeCheck();
63
64 uint8_t chans = 0;
65 volatile uint32_t* buf = getAnalogBuffer(hadc,&chans);
66 if(buf == NULL)
67 return;
68
69 for(AdcHandler* c : AdcHandler::adcHandlers){
70 c->adcUpd(buf,chans,hadc);
71 }
72}

301void tud_hid_set_report_cb(uint8_t itf, uint8_t report_id, hid_report_type_t report_type, uint8_t const* buffer, uint16_t bufsize){
302 if(report_type == HID_REPORT_TYPE_INVALID && report_id == 0){
303 report_id = *buffer;
304 }
305
306 if(UsbHidHandler::globalHidHandler!=nullptr)
307 UsbHidHandler::globalHidHandler->hidOut(report_id,report_type,buffer,bufsize);
308
309 if(report_id == HID_ID_HIDCMD){
310 if(HID_CommandInterface::globalInterface != nullptr)
311 HID_CommandInterface::globalInterface->hidCmdCallback((HID_CMD_Data_t*)(buffer));
312 }
