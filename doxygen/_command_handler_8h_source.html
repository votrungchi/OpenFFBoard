195 inline CommandStatus handleGetSet(const ParsedCommand& cmd,std::vector<CommandReply>& replies,TVal& value){
196 if(cmd.type == CMDtype::set){
197 value = static_cast<TVal>(cmd.val);
198 }else if(cmd.type == CMDtype::get){
199 replies.push_back(CommandReply(value));
200 }else{
201 return CommandStatus::ERR;
202 }
203 return CommandStatus::OK;
204 }

248 template<typename ID>
249 void registerCommand(const char* cmd,const ID cmdid,const char* help=nullptr,uint32_t flags = 0){
250 for(CmdHandlerCommanddef& cmdDef : registeredCommands){
251 if(cmdDef.cmdId == static_cast<uint32_t>(cmdid))
252 return; //already present
253 }
254
255 CmdHandlerCommanddef cmddef = {
256 .cmd=cmd,
257 .helpstring = help,
258 .cmdId = static_cast<uint32_t>(cmdid),
259 .flags = flags
260 };
261 this->registeredCommands.push_back(cmddef);
262 this->registeredCommands.shrink_to_fit();
263 }
