67/*
68If the metric is less than CP Offset - Dead Band, then the resulting force is given by the following formula:
69 force = Negative Coefficient * (q - (CP Offset – Dead Band))
70Similarly, if the metric is greater than CP Offset + Dead Band, then the resulting force is given by the
71following formula:
72 force = Positive Coefficient * (q - (CP Offset + Dead Band))
73A spring condition uses axis position as the metric.
74A damper condition uses axis velocity as the metric.
75An inertia condition uses axis acceleration as the metric.
76
77If the number of Condition report blocks is equal to the number of axes for the effect, then the first report
78block applies to the first axis, the second applies to the second axis, and so on. For example, a two-axis
79spring condition with CP Offset set to zero in both Condition report blocks would have the same effect as
80the joystick self-centering spring. When a condition is defined for each axis in this way, the effect must
81not be rotated.
82
83If there is a single Condition report block for an effect with more than one axis, then the direction along
84which the parameters of the Condition report block are in effect is determined by the direction parameters
85passed in the Direction field of the Effect report block. For example, a friction condition rotated 45
86degrees (in polar coordinates) would resist joystick motion in the northeast-southwest direction but would
87have no effect on joystick motion in the northwest-southeast direction.
88 */

90/*
91 * Calculates the resulting torque for FFB effects
92 * Takes current position input scaled from -0x7fff to 0x7fff
93 * Outputs a torque value from -0x7fff to 0x7fff (not yet clipped)
94 */
95void EffectsCalculator::calculateEffects(std::vector<std::unique_ptr<Axis>> &axes)
96{
97 for (auto &axis : axes) {
98 axis->calculateAxisEffects(isActive());
99 }
100
101 if(!isActive()){
102 return;
103 }
104
105 int32_t forceX = 0;
106 int32_t forceY = 0;
107 int32_t forceVector = 0;
108 uint8_t axisCount = axes.size();
109 bool validY = axisCount > 1;
110#if MAX_AXIS == 3
111 int32_t forceZ = 0;
112 bool validZ = axisCount > 2;
113#endif
114
115 for (uint8_t i = 0; i < MAX_EFFECTS; i++)
116 {
117 FFB_Effect *effect = &effects[i];
118
119 // Effect activated and not infinite (0 or 0xffff)
120 if (effect->state != EFFECT_STATE_INACTIVE && effect->duration != FFB_EFFECT_DURATION_INFINITE && effect->duration != 0){
121 // Start delay not yet reached
122 if(HAL_GetTick() < effect->startTime){
123 continue;
124 }
125 // If effect has expired make inactive
126 if (HAL_GetTick() > effect->startTime + effect->duration)
127 {
128 effect->state = EFFECT_STATE_INACTIVE;
129 }
130 }
131
132 // Filter out inactive effects
133 if (effect->state == EFFECT_STATE_INACTIVE)
134 {
135 continue;
136 }
137
138
139 //if (effect->conditionsCount == 0) {
140 forceVector = calcNonConditionEffectForce(effect);
141 //}
142
143 uint8_t directionEnableMask = this->directionEnableMask ? this->directionEnableMask : DIRECTION_ENABLE(axisCount);
144
145 if (effect->enableAxis & directionEnableMask || (effect->enableAxis & X_AXIS_ENABLE))
146 {
147 forceX += calcComponentForce(effect, forceVector, axes, 0);
148 forceX = clip<int32_t, int32_t>(forceX, -0x7fff, 0x7fff); // Clip
149 }
150 if (validY && (effect->enableAxis & directionEnableMask || (effect->enableAxis & Y_AXIS_ENABLE)))
151 {
152 forceY += calcComponentForce(effect, forceVector, axes, 1);
153 forceY = clip<int32_t, int32_t>(forceY, -0x7fff, 0x7fff); // Clip
154 }
155
156 }
157
158 axes[0]->setEffectTorque(forceX);
159 if (validY)
160 {
161 axes[1]->setEffectTorque(forceY);
162 }
163}

169int32_t EffectsCalculator::calcNonConditionEffectForce(FFB_Effect *effect) {
170 int32_t force_vector = 0;
171 int32_t magnitude = effect->magnitude;
172
173 // If using an envelope modulate the magnitude based on time
174 if(effect->useEnvelope){
175 magnitude = getEnvelopeMagnitude(effect);
176 }
177 switch (effect->type){
178
179 case FFB_EFFECT_CONSTANT:
180 { // Constant force is just the force
181 force_vector = ((int32_t)magnitude * (int32_t)(1 + effect->gain)) >> 8;
182
183 break;
184 }
185
186 case FFB_EFFECT_RAMP:
187 {
188 uint32_t elapsed_time = HAL_GetTick() - effect->startTime;
189 int32_t duration = effect->duration;
190 float force = (int32_t)effect->startLevel + ((int32_t)elapsed_time * (effect->endLevel - effect->startLevel)) / duration;
191 force_vector = (int32_t)(force * (1 + effect->gain)) >> 8;
192 break;
193 }
194
195 case FFB_EFFECT_SQUARE:
196 {
197 uint32_t elapsed_time = HAL_GetTick() - effect->startTime;
198 int32_t force = ((elapsed_time + effect->phase) % ((uint32_t)effect->period + 2)) < (uint32_t)(effect->period + 2) / 2 ? -magnitude : magnitude;
199 force_vector = force + effect->offset;
200 break;
201 }
202
203 case FFB_EFFECT_TRIANGLE:
204 {
205 int32_t force = 0;
206 int32_t offset = effect->offset;
207 uint32_t elapsed_time = HAL_GetTick() - effect->startTime;
208 uint32_t phase = effect->phase;
209 uint32_t period = effect->period;
210 float periodF = period;
211
212 int32_t maxMagnitude = offset + magnitude;
213 int32_t minMagnitude = offset - magnitude;
214 uint32_t phasetime = (phase * period) / 35999;
215 uint32_t timeTemp = elapsed_time + phasetime;
216 float remainder = timeTemp % period;
217 float slope = ((maxMagnitude - minMagnitude) * 2) / periodF;
218 if (remainder > (periodF / 2))
219 force = slope * (periodF - remainder);
220 else
221 force = slope * remainder;
222 force += minMagnitude;
223 force_vector = force;
224 break;
225 }
226
227 case FFB_EFFECT_SAWTOOTHUP:
228 {
229 float offset = effect->offset;
230 uint32_t elapsed_time = HAL_GetTick() - effect->startTime;
231 uint32_t phase = effect->phase;
232 uint32_t period = effect->period;
233 float periodF = effect->period;
234
235 float maxMagnitude = offset + magnitude;
236 float minMagnitude = offset - magnitude;
237 int32_t phasetime = (phase * period) / 35999;
238 uint32_t timeTemp = elapsed_time + phasetime;
239 float remainder = timeTemp % period;
240 float slope = (maxMagnitude - minMagnitude) / periodF;
241 force_vector = (int32_t)(minMagnitude + slope * (period - remainder));
242 break;
243 }
244
245 case FFB_EFFECT_SAWTOOTHDOWN:
246 {
247 float offset = effect->offset;
248 uint32_t elapsed_time = HAL_GetTick() - effect->startTime;
249 float phase = effect->phase;
250 uint32_t period = effect->period;
251 float periodF = effect->period;
252
253 float maxMagnitude = offset + magnitude;
254 float minMagnitude = offset - magnitude;
255 int32_t phasetime = (phase * period) / 35999;
256 uint32_t timeTemp = elapsed_time + phasetime;
257 float remainder = timeTemp % period;
258 float slope = (maxMagnitude - minMagnitude) / periodF;
259 force_vector = (int32_t)(minMagnitude + slope * (remainder)); // reverse time
260 break;
261 }
262
263 case FFB_EFFECT_SINE:
264 {
265 float t = HAL_GetTick() - effect->startTime;
266 float freq = 1.0f / (float)(std::max<uint16_t>(effect->period, 2));
267 float phase = (float)effect->phase / (float)35999; //degrees
268 float sine = sinf(2.0 * M_PI * (t * freq + phase)) * magnitude;
269 force_vector = (int32_t)(effect->offset + sine);
270 break;
271 }
272 default:
273 return 0;
274 break;
275 }
276
277 return force_vector;
278}

296int32_t EffectsCalculator::calcComponentForce(FFB_Effect *effect, int32_t forceVector, std::vector<std::unique_ptr<Axis>> &axes, uint8_t axis)
297{
298 int32_t result_torque = 0;
299 uint16_t direction;
300 uint8_t con_idx = 0; // condition block index
301
302 metric_t *metrics = axes[axis]->getMetrics();
303 uint8_t axisCount = axes.size();
304 float scaleSpeed = 40;//axes[axis]->getSpeedScalerNormalized(); // TODO decide if scalers are useful or not
305 float scaleAccel = 40;//axes[axis]->getAccelScalerNormalized();
306 uint8_t directionEnableMask = this->directionEnableMask ? this->directionEnableMask : DIRECTION_ENABLE(axisCount);
307 if (effect->enableAxis & directionEnableMask)
308 {
309 direction = effect->directionX;
310// if (effect->conditionsCount > 1)
311// {
312 con_idx = axis;
313// }
314 }
315 else
316 {
317 direction = axis == 0 ? effect->directionX : effect->directionY;
318 con_idx = axis;
319 }
320
321 //bool useForceDirectionForConditionEffect = (effect->enableAxis == DIRECTION_ENABLE && axisCount > 1 && effect->conditionsCount == 1);
322 bool rotateConditionForce = (axisCount > 1); // && effect->conditionsCount < axisCount
323 float angle = ((float)direction * (2*M_PI) / 36000.0);
324 float angle_ratio = axis == 0 ? sin(angle) : -1 * cos(angle);
325 angle_ratio = rotateConditionForce ? angle_ratio : 1.0;
326
327 switch (effect->type)
328 {
329 case FFB_EFFECT_CONSTANT:
330 {
331 // Optional filtering to reduce spikes
332 if (cfFilter_f < calcfrequency / 2 && cfFilter_f != 0 )
333 {
334 result_torque = effect->filter[con_idx]->process(forceVector);
335 }
336 }
337 case FFB_EFFECT_RAMP:
338 case FFB_EFFECT_SQUARE:
339 case FFB_EFFECT_TRIANGLE:
340 case FFB_EFFECT_SAWTOOTHUP:
341 case FFB_EFFECT_SAWTOOTHDOWN:
342 case FFB_EFFECT_SINE:
343 {
344 result_torque = -forceVector * angle_ratio;
345 break;
346 }
347
348 case FFB_EFFECT_SPRING:
349 {
350 float pos = metrics->pos;
351 result_torque -= calcConditionEffectForce(effect, pos, gain.spring, con_idx, spring_scaler, angle_ratio);
352 break;
353 }
354
355
368 case FFB_EFFECT_FRICTION: // TODO sometimes unstable.
369 {
370 float speed = metrics->speed * scaleSpeed;//effect->filter[con_idx]->process()
371
372 int16_t offset = effect->conditions[con_idx].cpOffset;
373 int16_t deadBand = effect->conditions[con_idx].deadBand;
374 int32_t force = 0;
375
376 // Effect is only active outside deadband + offset
377 if (abs((int32_t)speed - offset) > deadBand){
378
379 // remove offset/deadband from metric to compute force
380 speed -= (offset + (deadBand * (speed < offset ? -1 : 1)) );
381
382 // check if speed is in the 0..x% to rampup, if is this range, apply a sinusoidale function to smooth the torque (slow near 0, slow around the X% rampup
383 float rampupFactor = 1.0;
384 if (fabs (speed) < speedRampupPct) { // if speed in the range to rampup we apply a sinus curbe to ramup
385
386 float phaseRad = M_PI * ((fabs (speed) / speedRampupPct) - 0.5);// we start to compute the normalized angle (speed / normalizedSpeed@5%) and translate it of -1/2PI to translate sin on 1/2 periode
387 rampupFactor = ( 1 + sin(phaseRad ) ) / 2; // sin value is -1..1 range, we translate it to 0..2 and we scale it by 2
388
389 }
390
391 int8_t sign = speed >= 0 ? 1 : -1;
392 uint16_t coeff = speed < 0 ? effect->conditions[con_idx].negativeCoefficient : effect->conditions[con_idx].positiveCoefficient;
393 force = coeff * rampupFactor * sign;
394
395 //if there is a saturation, used it to clip result
396 if (effect->conditions[con_idx].negativeSaturation !=0 || effect->conditions[con_idx].positiveSaturation !=0) {
397 force = clip<int32_t, int32_t>(force, -effect->conditions[con_idx].negativeSaturation, effect->conditions[con_idx].positiveSaturation);
398 }
399
400// static int32_t last_force = 0;
401//
402// // if there is 2 successive torque with a different direction, we ignore the first one to remove oscillation
403// if (last_force * force >=0) {
404// force = ((gain.friction + 1) * force) >> 7;
405// result_torque -= force * angle_ratio;
406// }
407// last_force = force;
408 result_torque -= effect->filter[con_idx]->process( (((gain.friction + 1) * force) >> 8) * angle_ratio * friction_scaler);
409 }
410// float accel = metrics->accel * scaleAccel;
411// result_torque -= calcConditionEffectForce(effect, accel, gain.friction, con_idx, friction_scaler, angle_ratio);
412
413 break;
414 }
415 case FFB_EFFECT_DAMPER:
416 {
417
418 float speed = metrics->speed * scaleSpeed;
419 result_torque -= effect->filter[con_idx]->process(calcConditionEffectForce(effect, speed, gain.damper, con_idx, damper_scaler, angle_ratio));
420
421 break;
422 }
423
424 case FFB_EFFECT_INERTIA:
425 {
426 float accel = metrics->accel* scaleAccel;
427 result_torque -= effect->filter[con_idx]->process(calcConditionEffectForce(effect, accel, gain.inertia, con_idx, inertia_scaler, angle_ratio)); // Bump *60 the inertia feedback
428
429 break;
430 }
431
432 default:
433 // Unsupported effect
434 break;
435 }
436 return (result_torque * (global_gain+1)) >> 8; // Apply global gain
437}

444int32_t EffectsCalculator::calcConditionEffectForce(FFB_Effect *effect, float metric, uint8_t gain,
445 uint8_t idx, float scale, float angle_ratio)
446{
447 int16_t offset = effect->conditions[idx].cpOffset;
448 int16_t deadBand = effect->conditions[idx].deadBand;
449 int32_t force = 0;
450 float gainfactor = (float)(gain+1) / 256.0;
451
452 // Effect is only active outside deadband + offset
453 if (abs(metric - offset) > deadBand){
454 float coefficient = effect->conditions[idx].negativeCoefficient;
455 if(metric > offset){
456 coefficient = effect->conditions[idx].positiveCoefficient;
457 }
458 coefficient /= 0x7fff; // rescale the coefficient of effect
459
460 // remove offset/deadband from metric to compute force
461 metric = metric - (offset + (deadBand * (metric < offset ? -1 : 1)) );
462
463 force = clip<int32_t, int32_t>((coefficient * gainfactor * scale * (float)(metric)),
464 -effect->conditions[idx].negativeSaturation,
465 effect->conditions[idx].positiveSaturation);
466 }
467
468
469 return force * angle_ratio;
470}

478int32_t EffectsCalculator::getEnvelopeMagnitude(FFB_Effect *effect)
479{
480 if(effect->duration == FFB_EFFECT_DURATION_INFINITE || effect->duration == 0){
481 return effect->magnitude; // Effect is infinite. envelope is invalid
482 }
483 int32_t scaler = abs(effect->magnitude);
484 uint32_t elapsed_time = HAL_GetTick() - effect->startTime;
485 if (elapsed_time < effect->attackTime && effect->attackTime != 0)
486 {
487 scaler = (scaler - effect->attackLevel) * elapsed_time;
488 scaler /= (int32_t)effect->attackTime;
489 scaler += effect->attackLevel;
490 }
491 if (elapsed_time > (effect->duration - effect->fadeTime) && effect->fadeTime != 0)
492 {
493 scaler = (scaler - effect->fadeLevel) * (effect->duration - elapsed_time); // Reversed
494 scaler /= (int32_t)effect->fadeTime;
495 scaler += effect->fadeLevel;
496 }
497 scaler = signbit(effect->magnitude) ? -scaler : scaler; // Follow original sign of magnitude because envelope has no sign (important for constant force)
498 return scaler;
499}

501void EffectsCalculator::setFilters(FFB_Effect *effect){
502
503 std::function<void(std::unique_ptr<Biquad> &)> fnptr = [=](std::unique_ptr<Biquad> &filter){};
504
505 switch (effect->type)
506 {
507 case FFB_EFFECT_DAMPER:
508 fnptr = [=](std::unique_ptr<Biquad> &filter){
509 if (filter != nullptr)
510 filter->setBiquad(BiquadType::lowpass, (float)damper_f / (float)calcfrequency, damper_q, (float)0.0);
511 else
512 filter = std::make_unique<Biquad>(BiquadType::lowpass, (float)damper_f / (float)calcfrequency, damper_q, (float)0.0);
513 };
514 break;
515 case FFB_EFFECT_FRICTION:
516 fnptr = [=](std::unique_ptr<Biquad> &filter){
517 if (filter != nullptr)
518 filter->setBiquad(BiquadType::lowpass, (float)friction_f / (float)calcfrequency, friction_q, (float)0.0);
519 else
520 filter = std::make_unique<Biquad>(BiquadType::lowpass, (float)friction_f / (float)calcfrequency, friction_q, (float)0.0);
521 };
522 break;
523 case FFB_EFFECT_INERTIA:
524 fnptr = [=](std::unique_ptr<Biquad> &filter){
525 if (filter != nullptr)
526 filter->setBiquad(BiquadType::lowpass, (float)inertia_f / (float)calcfrequency, inertia_q, (float)0.0);
527 else
528 filter = std::make_unique<Biquad>(BiquadType::lowpass, (float)inertia_f / (float)calcfrequency, inertia_q, (float)0.0);
529 };
530 break;
531 case FFB_EFFECT_CONSTANT:
532 fnptr = [=](std::unique_ptr<Biquad> &filter){
533 if (filter != nullptr)
534 filter->setBiquad(BiquadType::lowpass, (float)cfFilter_f / (float)calcfrequency, cfFilter_qfloatScaler * (cfFilter_q+1), (float)0.0);
535 else
536 filter = std::make_unique<Biquad>(BiquadType::lowpass, (float)cfFilter_f / (float)calcfrequency, cfFilter_qfloatScaler * (cfFilter_q+1), (float)0.0);
537 };
538 break;
539 }
540
541
542 for (int i=0; i<MAX_AXIS; i++) {
543 fnptr(effect->filter[i]);
544 }
545}

562/*
563 * Read parameters from flash and restore settings
564 */
565void EffectsCalculator::restoreFlash()
566{
567 uint16_t filter;
568 if (Flash_Read(ADR_CF_FILTER, &filter))
569 {
570 this->cfFilter_f = filter & 0x1FF;
571 this->cfFilter_q = (filter >> 9) & 0x7F;
572 }
573 setCfFilter(this->cfFilter_f,this->cfFilter_q);
574
575 uint16_t effects = 0;
576 if(Flash_Read(ADR_AXIS_EFFECTS1, &effects)){
577 gain.friction = (effects >> 8) & 0xff;
578 gain.inertia = (effects & 0xff);
579 }
580 if(Flash_Read(ADR_AXIS_EFFECTS2, &effects)){
581 gain.damper = (effects >> 8) & 0xff;
582 gain.spring = (effects & 0xff);
583 }
584
585}

587// Saves parameters to flash
588void EffectsCalculator::saveFlash()
589{
590 uint16_t cffilter = (cfFilter_f & 0x1FF) | ((cfFilter_q & 0x7F) << 9);
591 Flash_Write(ADR_CF_FILTER, cffilter);
592 uint16_t effects = gain.inertia | (gain.friction << 8);
593 Flash_Write(ADR_AXIS_EFFECTS1, effects);
594 effects = gain.spring | (gain.damper << 8);
595 Flash_Write(ADR_AXIS_EFFECTS2, effects);
596

652CommandStatus EffectsCalculator::command(const ParsedCommand& cmd,std::vector<CommandReply>& replies){
653 switch(static_cast<EffectsCalculator_commands>(cmd.cmdId)){
654
655 case EffectsCalculator_commands::ffbfiltercf:
656 if (cmd.type == CMDtype::get)
657 {
658 replies.push_back(CommandReply(cfFilter_f));
659 }
660 else if (cmd.type == CMDtype::set)
661 {
662 setCfFilter(cmd.val,this->cfFilter_q);
663 }
664 break;
665
666 case EffectsCalculator_commands::ffbfiltercf_q:
667 if (cmd.type == CMDtype::get)
668 {
669 replies.push_back(CommandReply(cfFilter_q));
670 }
671 else if (cmd.type == CMDtype::set)
672 {
673 setCfFilter(this->cfFilter_f,cmd.val);
674 }
675
676 break;
677
678 case EffectsCalculator_commands::effects:
