43void I2CPort::saveFlash(){
44 if(this->getCommandHandlerInfo()->instance != 0){
45 return; // Only first instance can save
46 }
47 uint16_t data = (this->speedPreset & 0b11); // 2 bits reserved. 1 used
48 Flash_Write(ADR_I2CCONF1, data);

89CommandStatus I2CPort::command(const ParsedCommand& cmd,std::vector<CommandReply>& replies){
90
91 switch(static_cast<I2CPort_commands>(cmd.cmdId)){
92
93 case I2CPort_commands::speed:
94 if(cmd.type == CMDtype::get){
95 replies.emplace_back(this->speedPreset);
96 }else if(cmd.type == CMDtype::set){
97 setSpeedPreset(cmd.val);
98 }else if(cmd.type == CMDtype::info){
99 for(uint8_t i = 0; i<SpeedNames.size();i++){
100 replies.emplace_back(SpeedNames[i] + ":" + std::to_string(i));
101 }
102 }
103 break;
104
105 default:
106 return CommandStatus::NOT_FOUND;
107 }
108 return CommandStatus::OK;
109}
