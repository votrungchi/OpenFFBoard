43void I2CPort::saveFlash(){
44 if(this->getCommandHandlerInfo()->instance != 0){
45 return; // Only first instance can save
46 }
47 uint16_t data = (this->speedPreset & 0b11); // 2 bits reserved. 1 used
48 Flash_Write(ADR_I2CCONF1, data);
49}

51void I2CPort::restoreFlash(){
52 if(this->getCommandHandlerInfo()->instance != 0){
53 return; // Only first instance can save
54 }
55 uint16_t data;
56 if(Flash_Read(ADR_I2CCONF1, &data)){
57 setSpeedPreset(data & 0b11);
58 }

88CommandStatus I2CPort::command(const ParsedCommand& cmd,std::vector<CommandReply>& replies){
89
90 switch(static_cast<I2CPort_commands>(cmd.cmdId)){
91
92 case I2CPort_commands::speed:
93 if(cmd.type == CMDtype::get){
94 replies.push_back(CommandReply(this->speedPreset));
95 }else if(cmd.type == CMDtype::set){
96 setSpeedPreset(cmd.val);
97 }else if(cmd.type == CMDtype::info){
98 for(uint8_t i = 0; i<SpeedNames.size();i++){
99 replies.push_back(CommandReply(SpeedNames[i] + ":" + std::to_string(i)));
100 }
101 }
102 break;
103
104 default:
105 return CommandStatus::NOT_FOUND;
106 }
107 return CommandStatus::OK;
108}
