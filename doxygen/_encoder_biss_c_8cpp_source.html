30

34

52void EncoderBissC::Run(){
53 while(true){
54 requestNewDataSem.Take(); // Wait until a position is requested
55 waitData = true;
56 spiPort.receive_DMA(spi_buf, bytes, this); // Receive next frame
57 this->WaitForNotification(); // Wait until DMA is finished
58
59 if(updateFrame()){
60 pos = newPos;
61 //handle multiturn
62 if(pos-lastPos > 1<<(lenghtDataBit-1)){
63 mtpos--;
64 }else if(lastPos-pos > 1<<(lenghtDataBit-1)){
65 mtpos++;
66 }
67
68 lastPos = pos;
69 }else{
70 numErrors++;
71 }
72 waitData = false;
73 if(useWaitSem)
74 waitForUpdateSem.Give();
75
76 }
77}

90void EncoderBissC::saveFlash(){

92 buf |= ((this->spiSpeed-1) & 0x3) << 5;
93 Flash_Write(ADR_BISSENC_CONF1, buf);

154__attribute__((optimize("-Ofast")))
155bool EncoderBissC::updateFrame(){
156
157
158 //Put data into 64bit int to enable easy shifting
159 uint64_t rxData64;
160
161 // 32b operation reduces time needed
162 rxData64 = (uint64_t)__REV(decod_buf[0]) << 32;
163 rxData64 |= (uint64_t)__REV(decod_buf[1]);
164
165 // sample of rxData64
166 // like this 1100000000000000100001100111010000000101110111100000000000000000
167 rxData64 <<= __builtin_clzll(rxData64); // slice rxData to have a value starting with 1
168 rxData64 &= 0x3FFFFFFFFFFFFFFF; // remove the 2 first bit
169
170 // remove the first 1, count how many digit stay in buffer after removing the 0, if there is more than 32 digits,
171 // keep only 32st (on the left)
172 // 32 because the format is : (1+1+lenghtDataBit+1+1+6) - Align bitstream to left (Startbit, CDS, 22-bit Position, Error, Warning, CRC)
173 //int nbBit = log2(rxData64)+1;
174 int nbBit = 64-__builtin_clzll(rxData64); // Much faster than log2
175 if ( nbBit >= ( lenghtDataBit + 10 ) ) {
176 rxData64 >>= nbBit-( lenghtDataBit + 10 );
177 }
178
179 uint8_t crcRx = rxData64 & 0x3F; //extract last 6-bit digits to get CRC
180 uint32_t dataRx = (rxData64 >> 6) & ((1<<(lenghtDataBit + 2)) - 1); //Shift out CRC, AND with 24-bit mask to get raw data (position, error, warning)
181 newPos = (dataRx >> 2) & ((1<<lenghtDataBit) - 1); //Shift out error and warning, AND with 22-bit mask to get position
182
183
184 uint8_t crc = 0; //CRC seed is 0b000000
185 crc = ((dataRx >> 30) & 0x03);
186 crc = tableCRC6n[((dataRx >> 24) & 0x3F) ^ crc];
187 crc = tableCRC6n[((dataRx >> 18) & 0x3F) ^ crc];
188 crc = tableCRC6n[((dataRx >> 12) & 0x3F) ^ crc];
189 crc = tableCRC6n[((dataRx >> 6) & 0x3F) ^ crc];
190 crc = tableCRC6n[((dataRx >> 0) & 0x3F) ^ crc];
191 crc = 0x3F & ~crc; //CRC is output inverted
192
193 bool crc_ok = crc == crcRx;
194 return crc_ok;
195}
