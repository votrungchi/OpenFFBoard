57void CanButtons::saveFlash(){
58 uint16_t conf1 = (btnnum-1) & 0x3F;
59 conf1 |= (invert & 0x1) << 6;
60 Flash_Write(ADR_CANBTN_CONF1, conf1);
61
62 uint16_t conf2 = canId & 0x7ff;

66void CanButtons::restoreFlash(){
67 uint16_t conf;
68 if(Flash_Read(ADR_CANBTN_CONF1, &conf)){
69 setBtnNum((conf & 0x3F) +1);
70 invert = (conf >> 6) & 0x1;
71 }
72 // CAN settings
73 if(Flash_Read(ADR_CANBTN_CONF2, &conf)){
74 canId = conf & 0x7ff;
75 }

78void CanButtons::setBtnNum(uint8_t num){
79 num = clip<uint8_t,uint8_t>(num, 1, 64); // up to 8 PCF8574 can be chained resulting in 64 buttons
80 this->btnnum = num;

82 mask = 0xffffffffffffffff;

99CommandStatus CanButtons::command(const ParsedCommand& cmd,std::vector<CommandReply>& replies){
100
101 switch(static_cast<CanButtons_commands>(cmd.cmdId)){
102
103 case CanButtons_commands::btnnum:
104 if(cmd.type == CMDtype::get){
105 replies.push_back(CommandReply(this->btnnum));
106 }else if(cmd.type == CMDtype::set){
107 setBtnNum(cmd.val);
108 }
109 break;
110 case CanButtons_commands::invert:
111 return handleGetSet(cmd, replies, this->invert);
112 break;
113
114 case CanButtons_commands::canid:
115 if(cmd.type == CMDtype::get){
116 replies.push_back(CommandReply(this->canId));
117 }else if(cmd.type == CMDtype::set){
118 canId = (cmd.val) & 0x7ff;
119 setupCanPort(); // Set can filter
120 }
121 break;
122
123 default:
124 return CommandStatus::NOT_FOUND;
125 }
126 return CommandStatus::OK;
127}
