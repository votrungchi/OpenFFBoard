97bool AxesManager::setAxisCount(int8_t count) {
98 if (!this->validAxisRange(count)) {
99 return false; // invalid number of axis
100 }
101 // Really need to use some form of mutex
102 Flash_Write(ADR_AXIS_COUNT, count);
103
104 while (count < axis_count) {
105 uint8_t pos = axis_count-1;
106 axes.erase(axes.begin()+pos);
107 axis_count--;
108 }
109 while (count > axis_count) {
110 axes.push_back( std::make_unique<Axis>('X'+axis_count, control) ); // Axis are indexed from 1-X to 3-Z
111 axis_count++;
112 }
113 control->update_disabled = false;
114
115 // Allocate axis value buffer
116 this->axisValues = std::vector<int32_t>(axis_count,0);
117
118 return true;
119}
